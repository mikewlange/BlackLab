17

19  import java.util.ArrayList;
20  import java.util.HashMap;
21  import java.util.List;
22  import java.util.Map;
23  import java.util.Map.Entry;
24  
25  import org.xml.sax.Attributes;
26  
27  import nl.inl.blacklab.index.DocIndexerXmlHandlers;
28  import nl.inl.blacklab.index.HookableSaxHandler.ContentCapturingHandler;
29  import nl.inl.blacklab.index.HookableSaxHandler.ElementHandler;
30  import nl.inl.blacklab.index.Indexer;
31  import nl.inl.blacklab.index.complex.ComplexFieldProperty;
32  import nl.inl.blacklab.index.complex.ComplexFieldUtil;

34  /**
35   * Index a FoLiA file for WhiteLab 2.0.
36   * For information about FoLiA, see http://proycon.github.io/folia/
37   * For information about WhiteLab, see https://github.com/Taalmonsters/WhiteLab
38   */
39  public class DocIndexerWhiteLab2 extends DocIndexerXmlHandlers {
40  
41  	String xmlid;
42  
43  	String wordform;
44  
45  	String pos;
46  
47  	String lemma;
48  
49  	String phonetic;
50  
51  	String speaker;
52  
53  	String beginTime;
54  
55  	String endTime;
56  
57  	List<String> ids;
58  
59  	boolean lemPosProblemReported = false;
60  
61  	int numPhonAnnotations = 0;
62  
63  	/**
64  	 * If we have 1 PoS annotation, use pos tags without a set
65  	 * attribute. If we have 2, we use pos tags with
66  	 * set="http://ilk.uvt.nl/folia/sets/frog-mbpos-cgn"
67  	 */
68  	int numPosAnnotations = 0;
69  
70  	/**
71  	 * If we have 1 lemma annotation, use lemma tags without a set
72  	 * attribute. If we have 2, we use pos tags with
73  	 * set="http://ilk.uvt.nl/folia/sets/frog-mblem-nl"
74  	 */
75  	int numLemmaAnnotations = 0;
76  
77  	/** Are we capturing the features inside a pos element? */
78  	boolean capturePosFeatures = false;
79  
80  	/** The features inside the current pos element, to be indexed as subproperties */
81  	Map<String, String> posFeatures = new HashMap<>();
82  
83  	public DocIndexerWhiteLab2(Indexer indexer, String fileName, Reader reader) {
84  		super(indexer, fileName, reader);
85  
86  		// Get handles to the default properties (the main one & punct)
87  		final ComplexFieldProperty propMain = getMainProperty();
88  		final ComplexFieldProperty propPunct = getPropPunct();
89  
90  		// Add some extra properties
91  		final ComplexFieldProperty propLemma = addProperty("lemma");
92  		final ComplexFieldProperty propPartOfSpeech = addProperty("pos");
93  		final ComplexFieldProperty propPhonetic = addProperty("phonetic");
94  		final ComplexFieldProperty propXmlid = addProperty("xmlid");
95  		final ComplexFieldProperty propSpeaker = addProperty("speaker");
96  		final ComplexFieldProperty propBeginTime = addProperty("begin_time");
97  		final ComplexFieldProperty propEndTime = addProperty("end_time");
98  
99  		// Doc element: the individual documents to index
100 		addHandler("/FoLiA", new DocumentElementHandler());
101 
102 		// PoS annotation metadata: see which annotation we need to use.
103 		addHandler("pos-annotation", new ElementHandler() {
104 			@Override
105 			public void startElement(String uri, String localName,
106 					String qName, Attributes attributes) {
107 				super.startElement(uri, localName, qName, attributes);
108 				numPosAnnotations++;
109 			}
110 		});
111 
112 		// Lemma annotation metadata: see which annotation we need to use.
113 		addHandler("lemma-annotation", new ElementHandler() {
114 			@Override
115 			public void startElement(String uri, String localName,
116 					String qName, Attributes attributes) {
117 				super.startElement(uri, localName, qName, attributes);
118 				numLemmaAnnotations++;
119 			}
120 		});
121 
122 		addHandler("phon-annotation", new ElementHandler() {
123 			@Override
124 			public void startElement(String uri, String localName,
125 					String qName, Attributes attributes) {
126 				super.startElement(uri, localName, qName, attributes);
127 				numPhonAnnotations++;
128 			}
129 		});
130 
131 		// Word elements: index as main contents
132 		addHandler("w", new WordHandlerBase() {
133 
134 			@Override
135 			public void startElement(String uri, String localName, String qName,
136 					Attributes attributes) {
137 				super.startElement(uri, localName, qName, attributes);
138 				xmlid = attributes.getValue("xml:id");
139 				wordform = "";
140 				pos = "";
141 				lemma = "";
142 				phonetic = "";
143 			}
144 
145 			@Override
146 			public void endElement(String uri, String localName, String qName) {
147 				super.endElement(uri, localName, qName);
148 				if (wordform.length() > 0) {
149 
150 					propMain.addValue(wordform);
151 					propXmlid.addValue(xmlid.split("\\.", 2)[1]);
152 					ids.add(xmlid);
153 					propPartOfSpeech.addValue(pos);
154 					for (Entry<String, String> e: posFeatures.entrySet()) {
155 						// Add the separate PoS features as extra values at this position.
156 						String sep = ComplexFieldUtil.SUBPROPERTY_SEPARATOR;
157 						propPartOfSpeech.addValue(sep + e.getKey() + sep + e.getValue(), 0);
158 					}
159 					propLemma.addValue(lemma);
160 
161 					if (pos.length() == 0 || lemma.length() == 0) {
162 						if (!lemPosProblemReported) {
163 							lemPosProblemReported = true;
164 							System.err.println(
165 								"Word without Pos (set=http://ilk.uvt.nl/folia/sets/frog-mbpos-cgn) and/or lemma (set=http://ilk.uvt.nl/folia/sets/frog-mblem-nl "
166 								+ "at " + describePosition());
167 						}
168 					}
169 
170 					if (numPhonAnnotations > 0)
171 						propPhonetic.addValue(phonetic);
172 					else
173 						propPhonetic.addValue("");
174 
175 					if (speaker != null)
176 						propSpeaker.addValue(speaker);
177 					else
178 						propSpeaker.addValue("");
179 
180 					propPunct.addValue(" ");
181 				} else {
182 					ids.add("");
183 				}
184 			}
185 		});
186 
187 		// lemma element: contains lemma
188 		addHandler("lemma", new ElementHandler() {
189 			@Override
190 			public void startElement(String uri, String localName, String qName,
191 					Attributes attributes) {
192 				super.startElement(uri, localName, qName, attributes);
193 				String set = attributes.getValue("set");
194 				boolean isSetLess = set == null || set.length() == 0;
195 				boolean isFrog = !isSetLess && set.equals("http://ilk.uvt.nl/folia/sets/frog-mblem-nl");
196 				if (numLemmaAnnotations == 2 && isFrog ||
197 					numLemmaAnnotations == 1 && isSetLess) {
198 					// If there were 2 lemma annotation meta declarations,
199 					// we should use the frog ones; if only 1, the ones
200 					// without a "set" attribute.
201 					lemma = attributes.getValue("class");
202 					if (lemma == null)
203 						lemma = "";
204 				}
205 			}
206 		});
207 
208 		// pos element: contains part of speech
209 		addHandler("pos", new ElementHandler() {
210 			@Override
211 			public void startElement(String uri, String localName, String qName,
212 					Attributes attributes) {
213 				super.startElement(uri, localName, qName, attributes);
214 				String set = attributes.getValue("set");
215 				boolean isSetLess = set == null || set.length() == 0;
216 				boolean isFrog = !isSetLess && set.equals("http://ilk.uvt.nl/folia/sets/frog-mbpos-cgn");
217 				if (numPosAnnotations == 2 && isFrog ||
218 					numPosAnnotations == 1 && isSetLess) {
219 					// If there were 2 pos annotation meta declarations,
220 					// we should use the frog ones; if only 1, the ones
221 					// without a "set" attribute.
222 					pos = attributes.getValue("class");
223 					if (pos == null)
224 						pos = "";
225 					// Capture the features inside this pos element (and add the head PoS now)
226 					capturePosFeatures = true;
227 					posFeatures.clear();
228 					String posHead = pos.contains("(") ? pos.substring(0, pos.indexOf("(")).trim() : pos;
229 					posFeatures.put("head", posHead);
230 				}
231 			}
232 
233 			@Override
234 			public void endElement(String uri, String localName, String qName) {
235 				capturePosFeatures = false; // we have all the features
236 				super.endElement(uri, localName, qName);
237 			}
238 
239 		});
240 
241 		// pos/feat element: contains a part of speech feature
242 		addHandler("pos/feat", new ElementHandler() {
243 			@Override
244 			public void startElement(String uri, String localName, String qName,
245 					Attributes attributes) {
246 				super.startElement(uri, localName, qName, attributes);
247 				if (capturePosFeatures) {
248 					String featSubset = attributes.getValue("subset");
249 					String featClass = attributes.getValue("class");
250 					posFeatures.put(featSubset, featClass);
251 				}
252 			}
253 		});
254 
255 		// ph = phonetic
256 		addHandler("ph", new ContentCapturingHandler() {
257 			@Override
258 			public void startElement(String uri, String localName, String qName,
259 					Attributes attributes) {
260 				super.startElement(uri, localName, qName, attributes);
261 			}
262 
263 			@Override
264 			public void endElement(String uri, String localName, String qName) {
265 				super.endElement(uri, localName, qName);
266 				phonetic = getElementContent();
267 			}
268 		});
269 
270 		addHandler("timesegment", new ContentCapturingHandler() {
271 			@Override
272 			public void startElement(String uri, String localName, String qName,
273 					Attributes attributes) {
274 				super.startElement(uri, localName, qName, attributes);
275 				beginTime = attributes.getValue("begintime");
276 				endTime = attributes.getValue("endtime");
277 			}
278 
279 			@Override
280 			public void endElement(String uri, String localName, String qName) {
281 				super.endElement(uri, localName, qName);
282 				beginTime = null;
283 				endTime = null;
284 			}
285 		});
286 
287 		addHandler("timesegment/wref", new ContentCapturingHandler() {
288 			@Override
289 			public void startElement(String uri, String localName, String qName,
290 					Attributes attributes) {
291 				super.startElement(uri, localName, qName, attributes);
292 				String wref = attributes.getValue("id");
293 				String wid = ids.remove(0);
294 				while (!wid.equals(wref)) {
295 					propBeginTime.addValue("", 1);
296 					propEndTime.addValue("", 1);
297 					wid = ids.remove(0);
298 				}
299 				propBeginTime.addValue(beginTime, 1);
300 				propEndTime.addValue(endTime, 1);
301 			}
302 		});
303 
304 		// t (token) element directly under w (word) element: contains the word form
305 		addHandler("w/t", new ContentCapturingHandler() {
306 
307 			/** Tokens with a class attribute are (usually?) the original scanned token before correction,
308 			 *  so we skip them */
309 			boolean isOcr;
310 
311 			@Override
312 			public void startElement(String uri, String localName, String qName,
313 					Attributes attributes) {
314 				super.startElement(uri, localName, qName, attributes);
315 				isOcr = attributes.getValue("class") != null;
316 			}
317 
318 			@Override
319 			public void endElement(String uri, String localName, String qName) {
320 				super.endElement(uri, localName, qName);
321 				if (!isOcr)
322 					wordform = getElementContent();
323 			}
324 		});
325 
326 		// Sentence tags: index as tags in the content
327 		addHandler("s", new InlineTagHandler() {
328 			@Override
329 			public void startElement(String uri, String localName, String qName,
330 					Attributes attributes) {
331 				speaker = attributes.getValue("speaker");
332 				ids = new ArrayList<>();
333 				if (insideElement())
334 			        super.startElement(uri, localName, qName, attributes);
335 			}
336 
337 			@Override
338 			public void endElement(String uri, String localName, String qName) {
339 				speaker = null;
340 				while (ids.size() > 0) {
341 					ids.remove(0);
342 					propBeginTime.addValue("");
343 					propEndTime.addValue("");
344 				}
345 				if (insideElement())
346 			        super.endElement(uri, localName, qName);
347 			}
348 		});
349 
350 		// Paragraph tags: index as tags in the content
351 		addHandler("p", new InlineTagHandler());
352 
353 		// <event/> tags: index as tags in the content
354 		addHandler("event", new InlineTagHandler());
355 
356 		// meta elements: metadata fields
357 		// [NOT USED FOR OPENSONAR..?]
358 		addHandler("meta", new ContentCapturingHandler() {
359 
360 
361 			private String metadataFieldName;
362 
363 			/** Open tag: add metadata field */
364 			@Override
365 			public void startElement(String uri, String localName, String qName, Attributes attributes) {
366 				super.startElement(uri, localName, qName, attributes);
367 				metadataFieldName = attributes.getValue("id");
368 			}
369 
370 			@Override
371 			public void endElement(String uri, String localName, String qName) {
372 				super.endElement(uri, localName, qName);
373 				if (metadataFieldName != null)
374 					addMetadataField(metadataFieldName, getElementContent());
375 			}
376 		});
377 	}

379 	public static void main(String[] args) {
380 		System.out.println("NL B".replaceAll("[\\s\\./]", "_"));
381 		System.out.println("NL/B".replaceAll("[\\s\\./]", "_"));
382 		System.out.println("a.b.c.d".replaceAll("[\\s\\./]", "_"));
383 	}
