40  /**
41   * Store string content by id in a directory of compound files with a TOC file. Quickly retrieve
42   * (parts of) the string content.
43   *
44   * This implementation stores the strings in UTF-8 encoding to save disk space. To guarantee
45   * reasonably fast random access times, we keep track of "block offsets", which are byte offsets to
46   * the start of (fixed char size) blocks. Block size in bytes can be slightly larger than char size
47   * because some UTF-8 characters take up more than 1 byte. If the block size is 1000 chars, block
48   * offsets might be [0, 1011, 2015, 3020].
49   */
50  public class ContentStoreDirUtf8 extends ContentStoreDirAbstract {
51  	private static final String CHAR_ENCODING = "UTF-8";
52  
53  	/** Table of contents entry */
54  	static class TocEntry {
55  		/** id of the string */
56  		public int id;
57  
58  		/** id of the file the string was stored in */
59  		public int fileId;
60  
61  		/** byte offset into the file of the string */
62  		public int entryOffsetBytes;
63  
64  		/** length of the string in bytes */
65  		public int entryLengthBytes;
66  
67  		/** length of the string in characters */
68  		public int entryLengthCharacters;
69  
70  		/** fixed block size in characters (note that byte size differs per block) */
71  		public int blockSizeCharacters;
72  
73  		/** relative block start offsets in bytes */
74  		public int[] blockOffsetBytes;
75  
76  		/** was this entry deleted? (can be removed in next compacting run) */
77  		public boolean deleted;
78  
79  		public TocEntry(int id, int fileId, int offset, int length, int charLength, int blockSize,
80  				boolean deleted, int[] blockOffset) {
81  			super();
82  			this.id = id;
83  			this.fileId = fileId;
84  			entryOffsetBytes = offset;
85  			entryLengthBytes = length;
86  			entryLengthCharacters = charLength;
87  			blockSizeCharacters = blockSize;
88  			this.deleted = deleted;
89  			blockOffsetBytes = blockOffset;
90  		}

92  		/**
93  		 * Store TOC entry in the TOC file
94  		 *
95  		 * @param buf
96  		 *            where to serialize to
97  		 * @throws IOException
98  		 *             on error
99  		 */
100 		public void serialize(ByteBuffer buf) throws IOException {
101 			buf.putInt(id);
102 			buf.putInt(fileId);
103 			buf.putInt(entryOffsetBytes);
104 			buf.putInt(entryLengthBytes);
105 			buf.putInt(deleted ? -1 : entryLengthCharacters);
106 			buf.putInt(blockSizeCharacters);
107 			buf.putInt(blockOffsetBytes.length);
108 			IntBuffer ib = buf.asIntBuffer();
109 			ib.put(blockOffsetBytes);
110 			buf.position(buf.position() + blockOffsetBytes.length * Integer.SIZE / Byte.SIZE);
111 		}
112 
113 		/**
114 		 * Read TOC entry from the TOC file
115 		 *
116 		 * @param buf
117 		 *            the buffer to read from
118 		 * @return new TocEntry
119 		 * @throws IOException
120 		 *             on error
121 		 */
122 		public static TocEntry deserialize(ByteBuffer buf) throws IOException {
123 			int id = buf.getInt();
124 			int fileId = buf.getInt();
125 			int offset = buf.getInt();
126 			int length = buf.getInt();
127 			int charLength = buf.getInt();
128 			boolean deleted = charLength < 0;
129 			int blockSize = buf.getInt();
130 			int nBlocks = buf.getInt();
131 			int[] blockOffsetBytes = new int[nBlocks];
132 			IntBuffer ib = buf.asIntBuffer();
133 			ib.get(blockOffsetBytes);
134 			buf.position(buf.position() + blockOffsetBytes.length * Integer.SIZE / Byte.SIZE);
135 			return new TocEntry(id, fileId, offset, length, charLength, blockSize, deleted,
136 					blockOffsetBytes);
137 		}
138 
139 		/**
140 		 * Get the offset of the first byte of the specified block.
141 		 *
142 		 * @param blockNumber
143 		 *            which block?
144 		 * @return byte offset of the first byte in the file
145 		 */
146 		public long getBlockStartOffset(int blockNumber) {
147 			return entryOffsetBytes + blockOffsetBytes[blockNumber];
148 		}
149 
150 		/**
151 		 * Get the offset of the first byte beyond the specified block.
152 		 *
153 		 * @param blockNumber
154 		 *            which block?
155 		 * @return byte offset of the first byte beyond the block in the file
156 		 */
157 		public long getBlockEndOffset(int blockNumber) {
158 			if (blockNumber < blockOffsetBytes.length - 1)
159 				return entryOffsetBytes + blockOffsetBytes[blockNumber + 1];
160 			return entryOffsetBytes + entryLengthBytes;
161 		}
162 
163 		/**
164 		 * Size of this entry serialized
165 		 *
166 		 * @return the size in bytes
167 		 */
168 		public int sizeBytes() {
169 			return 28 + blockOffsetBytes.length * 4;
170 		}
171 	}

257 	/**
258 	 * What block size to use when adding a new document to the content store. Contributing factors
259 	 * for choosing block size:
260 	 * - larger blocks improve compression ratio
261 	 * - larger blocks decrease number of blocks you have to read
262 	 * - smaller blocks decrease the decompression time
263 	 * - smaller blocks increase the chance that we only have to read one disk block for a single concordance
264 	 *   (disk blocks are generally 2 or 4K)
265 	 * - consider OS I/O caching and memory mapping. Then it becomes the difference between reading
266 	 *   a few bytes from memory and reading a few kilobytes and decompressing them. Right now,
267 	 *   making concordances is often CPU-bound (because of decompression?)
268 	 */
269 	protected int newEntryBlockSizeCharacters = 4000;

292 	/**
293 	 * @param dir content store dir
294 	 * @param create if true, create a new content store
295 	 */
296 	public ContentStoreDirUtf8(File dir, boolean create) {
297 		this.dir = dir;
298 		if (!dir.exists())
299 			dir.mkdir();
300 		tocFile = new File(dir, "toc.dat");
301 		if (create && tocFile.exists()) {
302 			// Delete the ContentStore files
303 			tocFile.delete();
304 			new File(dir, "version.dat").delete();
305 			File[] dataFiles = dir.listFiles(new FilenameFilter() {
306 				@Override
307 				public boolean accept(File dir_, String name) {
308 					return name.matches("data\\d+.dat");
309 				}
310 			});
311 			for (File f : dataFiles) {
312 				f.delete();
313 			}
314 		}
315 		toc = Maps.mutable.empty();
316 		if (tocFile.exists())
317 			readToc();
318 		tocModified = false;
319 		if (create) {
320 			clear();
321 			if (tocFile.exists())
322 				tocFile.delete();
323 			setStoreType();
324 		}
325 		blockOffsetWhileStoring = new ArrayList<Integer>();
326 	}

337 	 * Delete all content in the document store
338 	 */
339 	@Override
340 	public void clear() {
341 		closeCurrentStoreFile();
342 
343 		// delete all data files and empty TOC
344 		for (Map.Entry<Integer, TocEntry> me : toc.entrySet()) {
345 			TocEntry e = me.getValue();
346 			File f = getContentFile(e.fileId);
347 			if (f.exists())
348 				f.delete();
349 		}
350 		toc.clear();
351 		tocModified = true;
352 		currentFileId = 1;
353 		currentFileLength = 0;
354 		nextId = 1;
355 	}

357 	private void mapToc(boolean writeable) throws IOException {
358 		tocRaf = new RandomAccessFile(tocFile, writeable ? "rw" : "r");
359 		long fl = tocFile.length();
360 		if (writeable) {
361 			fl += writeMapReserve;
362 		} // leave 1M room at the end
363 		tocFileChannel = tocRaf.getChannel();
364 		tocFileBuffer = tocFileChannel.map(writeable ? MapMode.READ_WRITE : MapMode.READ_ONLY, 0,
365 				fl);
366 	}
367 
368 	private void closeMappedToc() {
369 		if (tocFileBuffer == null)
370 			return; // not mapped
371 		try {
372 			tocFileChannel.close();
373 			tocFileChannel = null;
374 			tocRaf.close();
375 			tocRaf = null;
376 
377 			tocFileBuffer = null;
378 
379 		} catch (IOException e) {
380 			throw new RuntimeException(e);
381 		}
382 	}
383 
384 	/**
385 	 * Read the table of contents from the file
386 	 */
387 	private void readToc() {
388 		toc.clear();
389 		try {
390 			mapToc(false);
391 			try {
392 				tocFileBuffer.position(0);
393 				int n = tocFileBuffer.getInt();
394 				for (int i = 0; i < n; i++) {
395 					TocEntry e = TocEntry.deserialize(tocFileBuffer);
396 					toc.put(e.id, e);
397 
398 					// Keep track of the current content file (file with highest ID)
399 					if (e.fileId > currentFileId) {
400 						currentFileId = e.fileId;
401 						currentFileLength = 0;
402 					}
403 
404 					// Keep track of the length of the current content file
405 					if (e.fileId == currentFileId) {
406 						int endOfEntry = e.entryOffsetBytes + e.entryLengthBytes;
407 						if (endOfEntry > currentFileLength)
408 							currentFileLength = endOfEntry;
409 					}
410 
411 					// Keep track of what the next ID should be
412 					if (e.id + 1 > nextId)
413 						nextId = e.id + 1;
414 				}
415 			} finally {
416 				closeMappedToc();
417 			}
418 		} catch (Exception e) {
419 			throw ExUtil.wrapRuntimeException(e);
420 		}
421 	}
422 
423 	private void writeToc() {
424 		try {
425 			mapToc(true);
426 			tocFileBuffer.putInt(toc.size());
427 			try {
428 				for (TocEntry e : toc.values()) {
429 					if (tocFileBuffer.remaining() < e.sizeBytes()) {
430 						// Close and re-open with extra writing room
431 						int p = tocFileBuffer.position();
432 						closeMappedToc();
433 						mapToc(true);
434 						tocFileBuffer.position(p);
435 					}
436 					e.serialize(tocFileBuffer);
437 				}
438 			} finally {
439 				closeMappedToc();
440 			}
441 		} catch (Exception e) {
442 			throw ExUtil.wrapRuntimeException(e);
443 		}
444 		tocModified = false;
445 	}

452 		closeCurrentStoreFile();

478 	/**
479 	 * Encode and write the block we've compiled so far and reset for next block
480 	 * @param os where to write to block to
481 	 */
482 	public void writeCurrentBlock(OutputStream os) {
483 		try {
484 			// Encode and write block
485 
486 			// OPT: we could try memory-mapping the current content store file (not sure if
487 			// you can grow a file in size while memory-mapped? maybe just put it at a fixed
488 			// size?) for additional speed.
489 
490 			String blockContent = currentBlockContents.toString();
491 			if (blockContent.length() == 0)
492 				throw new RuntimeException("ERROR, tried to write an empty block");
493 			byte[] buf = encodeBlock(blockContent);
494 			os.write(buf);
495 			bytesWritten += buf.length;
496 			currentBlockContents = new StringBuilder(newEntryBlockSizeCharacters);
497 		} catch (IOException e) {
498 			throw new RuntimeException(e);
499 		}
500 	}
501 
502 	/**
503 	 * Store part of a piece of large content. This may be called several times to store chunks of
504 	 * content, but MUST be *finished* by calling the "normal" store() method. You may call store()
505 	 * with the empty string if you wish.
506 	 *
507 	 * @param content
508 	 *            the content to store
509 	 */
510 	@Override
511 	public synchronized void storePart(String content) {
512 		if (content.length() == 0)
513 			return;
514 		if (blockOffsetWhileStoring.size() == 0)
515 			blockOffsetWhileStoring.add(0); // first offset is always 0
516 
517 		// Calculate what charsFromEntryWritten will be after storing this part
518 		// (used to determine if we will cross a block boundary)
519 		int offsetAfterThisPart = charsFromEntryWritten + content.length();
520 
521 		// See if we're about to cross any block boundaries; if so, write the content up to the
522 		// first block boundary, save the new block offset, and repeat.
523 		int thisPartCharsWritten = 0, thisPartCharsLeftToWrite = content.length();
524 
525 		OutputStream os = openCurrentStoreFile();
526 		while (true) {
527 			// Will we cross a(nother) block boundary writing this part of the content?
528 			int nextBlockBoundary = blockOffsetWhileStoring.size() * newEntryBlockSizeCharacters;
529 			boolean willWeCrossBlockBoundary = (offsetAfterThisPart > nextBlockBoundary);
530 			if (!willWeCrossBlockBoundary) {
531 				// No; break out of this loop and write the last bit of content.
532 				break;
533 			}
534 
535 			// How many characters till the block boundary?
536 			int charsRemainingInBlock = nextBlockBoundary - charsFromEntryWritten;
537 
538 			// Are we at the block boundary already?
539 			if (charsRemainingInBlock > 0) {
540 				// No, not at the boundary yet; add last piece to block
541 				// and update bookkeeping variables
542 				addToBlock(content.substring(thisPartCharsWritten, thisPartCharsWritten
543 						+ charsRemainingInBlock));
544 				charsFromEntryWritten += charsRemainingInBlock;
545 
546 				thisPartCharsWritten += charsRemainingInBlock;
547 				thisPartCharsLeftToWrite -= charsRemainingInBlock;
548 			}
549 
550 			// We are now at a block boundary. Write the block and
551 			// store next block offset.
552 			if (currentBlockContents.length() > 0) {
553 				writeCurrentBlock(os);
554 				blockOffsetWhileStoring.add(bytesWritten);
555 			}
556 		}
557 		// No more block boundaries to cross. If there's any content left to write in the
558 		// current block,
559 		// do so now.
560 		if (thisPartCharsLeftToWrite > 0) {
561 			addToBlock(content.substring(thisPartCharsWritten, thisPartCharsWritten
562 					+ thisPartCharsLeftToWrite));
563 			charsFromEntryWritten += thisPartCharsLeftToWrite;
564 		}
565 	}

568 	 * Convert the String representation of a block to a byte buffer
569 	 *
570 	 * @param block
571 	 *            the block content
572 	 * @return the byte buffer representation
573 	 */
574 	protected byte[] encodeBlock(String block) {
575 		try {
576 			return block.getBytes(CHAR_ENCODING);
577 		} catch (UnsupportedEncodingException e) {
578 			throw new RuntimeException(e);
579 		}
580 	}

583 	 * Convert the byte buffer representation of a block back to the original String
584 	 *
585 	 * @param buf
586 	 *            the byte buffer
587 	 * @param offset
588 	 *            offset in the buffer
589 	 * @param length
590 	 *            length of the block in bytes
591 	 * @return the original String content
592 	 */
593 	protected String decodeBlock(byte[] buf, int offset, int length) {
594 		try {
595 			return new String(buf, offset, length, CHAR_ENCODING);
596 		} catch (UnsupportedEncodingException e) {
597 			throw new RuntimeException(e);
598 		}
599 	}

601 	/**
602 	 * Store the given content and assign an id to it
603 	 *
604 	 * @param content
605 	 *            the content to store
606 	 * @return the id assigned to the content
607 	 */
608 	@Override
609 	public synchronized int store(String content) {
610 		storePart(content);
611 		if (currentBlockContents.length() > 0) {
612 			// Write the last (not completely full) block
613 			OutputStream os = openCurrentStoreFile();
614 			writeCurrentBlock(os);
615 		}
616 		int[] blockOffsetArray = new int[blockOffsetWhileStoring.size()];
617 		int i = 0;
618 		for (Integer bo : blockOffsetWhileStoring) {
619 			blockOffsetArray[i] = bo;
620 			i++;
621 		}
622 		TocEntry e = new TocEntry(nextId, currentFileId, currentFileLength, bytesWritten,
623 				charsFromEntryWritten, newEntryBlockSizeCharacters, false, blockOffsetArray);
624 		nextId++;
625 		currentFileLength += bytesWritten;
626 		toc.put(e.id, e);
627 		tocModified = true;
628 		charsFromEntryWritten = 0;
629 		bytesWritten = 0;
630 		blockOffsetWhileStoring = new ArrayList<Integer>();
631 		return e.id;
632 	}
633 
634 	private OutputStream openCurrentStoreFile() {
635 		try {
636 			boolean createNew = false;
637 			if (currentFileLength > dataFileSizeHint) {
638 				// Current file is full!
639 				currentFileId++;
640 				currentFileLength = 0;
641 				if (currentStoreFileStream != null)
642 					currentStoreFileStream.close();
643 				currentStoreFileStream = null;
644 				createNew = true; // start a new store file; delete if old stuff hanging around
645 			}
646 			if (currentStoreFileStream == null) {
647 				File f = getContentFile(currentFileId);
648 				if (createNew && f.exists())
649 					f.delete(); // leftover from previous index; delete
650 				currentStoreFileStream = new BufferedOutputStream(new FileOutputStream(f, true));
651 			}
652 			return currentStoreFileStream;
653 		} catch (Exception e) {
654 			throw ExUtil.wrapRuntimeException(e);
655 		}
656 	}
657 
658 	private void closeCurrentStoreFile() {
659 		try {
660 			if (currentStoreFileStream != null)
661 				currentStoreFileStream.close();
662 		} catch (IOException e) {
663 			throw new RuntimeException(e);
664 		}
665 	}

692 	/**
693 	 * Retrieve one or more substrings from the specified content.
694 	 *
695 	 * This is more efficient than retrieving the whole content, or retrieving parts in separate
696 	 * calls, because the file is only opened once and random access is used to read only the
697 	 * required parts.
698 	 *
699 	 * NOTE: if offset and length are both -1, retrieves the whole content. This is used by the
700 	 * retrieve(id) method.
701 	 *
702 	 * @param contentId
703 	 *            id of the entry to get substrings from
704 	 * @param start
705 	 *            the starting points of the substrings (in characters).
706 	 *            -1 means "start of document"
707 	 * @param end
708 	 *            the end points of the substrings (in characters).
709 	 *            -1 means "end of document"
710 	 * @return the parts
711 	 */
712 	@Override
713 	public synchronized String[] retrieveParts(int contentId, int[] start, int[] end) {
714 		try {
715 			// Find the correct TOC entry
716 			TocEntry e = toc.get(contentId);
717 			if (e == null || e.deleted)
718 				return null;
719 
720 			// Sanity-check parameters
721 			int n = start.length;
722 			if (n != end.length)
723 				throw new RuntimeException("start and end must be of equal length");
724 
725 			// Create array for results
726 			String[] result = new String[n];
727 
728 			// Open the correct file
729 			FileInputStream fileInputStream = new FileInputStream(getContentFile(e.fileId));
730 			try {
731 				FileChannel fileChannel = fileInputStream.getChannel();
732 
733 				// Retrieve the strings requested
734 				for (int i = 0; i < n; i++) {
735 					int a = start[i];
736 					int b = end[i];
737 
738 //					if (a == -1 && b == -1) {
739 //						// This means "retrieve whole content"
740 //						a = 0;
741 //						b = e.entryLengthCharacters;
742 //					}
743 					if (a == -1)
744 						a = 0;
745 					if (b == -1)
746 						b = e.entryLengthCharacters;
747 
748 					// Check values
749 					if (a < 0 || b < 0) {
750 						throw new RuntimeException("Illegal values, start = " + a + ", end = " + b);
751 					}
752 					if (a > e.entryLengthCharacters || b > e.entryLengthCharacters) {
753 						throw new RuntimeException("Value(s) out of range, start = " + a
754 								+ ", end = " + b + ", content length = " + e.entryLengthCharacters);
755 					}
756 					if (b <= a) {
757 						throw new RuntimeException(
758 								"Tried to read empty or negative length snippet (from " + a
759 										+ " to " + b + ")");
760 					}
761 
762 					// 1 - determine what blocks to read
763 					int firstBlock = a / e.blockSizeCharacters;
764 					int lastBlock = (b - 1) / e.blockSizeCharacters;
765 
766 					// 2 - read and decode blocks
767 					StringBuilder decoded = new StringBuilder();
768 					for (int j = firstBlock; j <= lastBlock; j++) {
769 						long readStartOffset = e.getBlockStartOffset(j);
770 						int bytesToRead = (int) (e.getBlockEndOffset(j) - readStartOffset);
771 						ByteBuffer buffer = ByteBuffer.allocate(bytesToRead);
772 						int bytesRead = fileChannel.read(buffer, readStartOffset);
773 						if (bytesRead < bytesToRead) {
774 							// Apparently, something went wrong.
775 							throw new RuntimeException("Not enough bytes read, " + bytesRead
776 									+ " < " + bytesToRead);
777 						}
778 						decoded.append(decodeBlock(buffer.array(), 0, bytesRead));
779 					}
780 
781 					// 3 - take just what we need
782 					int firstChar = a % e.blockSizeCharacters;
783 					result[i] = decoded.toString().substring(firstChar, firstChar + b - a);
784 				}
785 			} finally {
786 				fileInputStream.close();
787 			}
788 			return result;
789 		} catch (Exception e) {
790 			throw ExUtil.wrapRuntimeException(e);
791 		}
792 	}
