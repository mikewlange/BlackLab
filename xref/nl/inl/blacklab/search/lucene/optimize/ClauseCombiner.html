3   import java.util.HashSet;
4   import java.util.Set;

6   import org.apache.lucene.index.IndexReader;
7   
8   import nl.inl.blacklab.search.lucene.BLSpanQuery;
9   
10  public abstract class ClauseCombiner {
11  
12  	public static final int CANNOT_COMBINE = Integer.MAX_VALUE;
13  
14  	public abstract int priority(BLSpanQuery left, BLSpanQuery right, IndexReader reader);
15  
16  	public abstract BLSpanQuery combine(BLSpanQuery left, BLSpanQuery right, IndexReader reader);
17  
18  	public boolean canCombine(BLSpanQuery left, BLSpanQuery right, IndexReader reader) {
19  		return priority(left, right, reader) != CANNOT_COMBINE;
20  	}
21  
22  	public static Set<ClauseCombiner> all(boolean nfa) {
23  		HashSet<ClauseCombiner> all = new HashSet<>();
24  		all.add(new ClauseCombinerRepetition());
25  		all.add(new ClauseCombinerInternalisation());
26  		all.add(new ClauseCombinerAnyExpansion());
27  		all.add(new ClauseCombinerNot());
28  		if (nfa)
29  			all.add(new ClauseCombinerNfa());
30  		return all;
31  	}
32  }
