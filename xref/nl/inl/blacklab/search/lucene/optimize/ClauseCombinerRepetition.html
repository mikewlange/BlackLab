1   package nl.inl.blacklab.search.lucene.optimize;
2   
3   import org.apache.lucene.index.IndexReader;
4   
5   import nl.inl.blacklab.search.lucene.BLSpanQuery;
6   import nl.inl.blacklab.search.lucene.SpanQueryRepetition;
7   
8   /**
9    * Recognize adjacent identical clauses and combine them.
10   */
11  class ClauseCombinerRepetition extends ClauseCombiner {
12  
13  	private static final int PRIORITY = 1;
14  
15  	@Override
16  	public int priority(BLSpanQuery left, BLSpanQuery right, IndexReader reader) {
17  		if (left.equals(right))
18  			return 1;
19  		BLSpanQuery leftCl = left instanceof SpanQueryRepetition ? ((SpanQueryRepetition)left).getClause() : left;
20  		BLSpanQuery rightCl = right instanceof SpanQueryRepetition ? ((SpanQueryRepetition)right).getClause() : right;
21  		return leftCl.equals(rightCl) ? PRIORITY : CANNOT_COMBINE;
22  	}
23  
24  	@Override
25  	public BLSpanQuery combine(BLSpanQuery left, BLSpanQuery right, IndexReader reader) {
26  		if (!canCombine(left, right, reader))
27  			throw new UnsupportedOperationException("Cannot combine " + left + " and " + right);
28  		BLSpanQuery leftCl = left;
29  		int leftMin = 1, leftMax = 1;
30  		int rightMin = 1, rightMax = 1;
31  		if (left instanceof SpanQueryRepetition) {
32  			SpanQueryRepetition l = ((SpanQueryRepetition)left);
33  			leftCl = l.getClause();
34  			leftMin = l.getMinRep();
35  			leftMax = l.getMaxRep();
36  		}
37  		if (right instanceof SpanQueryRepetition) {
38  			SpanQueryRepetition r = ((SpanQueryRepetition)right);
39  			rightMin = r.getMinRep();
40  			rightMax = r.getMaxRep();
41  		}
42  		return new SpanQueryRepetition(leftCl, leftMin + rightMin, BLSpanQuery.addMaxValues(leftMax, rightMax));
43  	}
44  }
