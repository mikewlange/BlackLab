38  /**
39   * Translates a TextPattern to a Lucene SpanQuery object.
40   */
41  public class TextPatternTranslatorSpanQuery extends TextPatternTranslator<SpanQuery> {
42  
43  	@Override
44  	public SpanQuery and(QueryExecutionContext context, List<SpanQuery> clauses) {
45  		return new SpanQueryAnd(clauses);
46  	}
47  
48  	@Override
49  	public SpanQuery andNot(QueryExecutionContext context, SpanQuery include, SpanQuery exclude) {
50  		return new SpanQueryPositionFilter(include, exclude, TextPatternPositionFilter.Operation.MATCHES, true);
51  	}
52  
53  	@Override
54  	public SpanQuery or(QueryExecutionContext context, List<SpanQuery> clauses) {
55  		return new BLSpanOrQuery(clauses.toArray(new SpanQuery[] {}));
56  	}
57  
58  	@Override
59  	public SpanQuery regex(QueryExecutionContext context, String value) {
60  		String valueNoStartEndMatch = value.replaceAll("\\^|\\$", "");
61  		return new BLSpanMultiTermQueryWrapper<RegexpQuery>(new RegexpQuery(
62  				new Term(context.luceneField(), context.optDesensitize(valueNoStartEndMatch))));
63  	}
64  
65  	@Override
66  	public SpanQuery sequence(QueryExecutionContext context, List<SpanQuery> clauses) {
67  		return new SpanQuerySequence(clauses);
68  	}
69  
70  	@Override
71  	public SpanQuery docLevelAnd(QueryExecutionContext context, List<SpanQuery> clauses) {
72  		return new SpanQueryDocLevelAnd(clauses);
73  	}
74  
75  	@Override
76  	public SpanQuery fuzzy(QueryExecutionContext context, String value, int maxEdits, int prefixLength) {
77  		return new SpanFuzzyQuery(new Term(context.luceneField(), context.optDesensitize(value)), maxEdits, prefixLength);
78  	}
79  
80  	@Override
81  	public SpanQuery tags(QueryExecutionContext context, String elementName, Map<String, String> attr) {
82  		SpanQuery allTags;
83  		if (context.tagLengthInPayload()) {
84  			// Modern index, with tag length in payload
85  			allTags = new SpanQueryTags(context, elementName);
86  		} else {
87  			// Older index, with end tags stored in separate property
88  			allTags = new SpanQueryTagsOld(context, elementName);
89  		}
90  		if (attr == null || attr.size() == 0)
91  			return allTags;
92  
93  		// Construct attribute filters
94  		List<SpanQuery> attrFilters = new ArrayList<SpanQuery>();
95  		QueryExecutionContext startTagContext = context.withProperty(ComplexFieldUtil.START_TAG_PROP_NAME);
96  		for (Map.Entry<String,String> e: attr.entrySet()) {
97  			String value = optInsensitive(context, "@" + e.getKey() + "__" + e.getValue());
98  			attrFilters.add(term(startTagContext, value));
99  		}
100 
101 		// Filter the tags
102 		// (NOTE: only works for start tags and full elements because attribute values
103 		//  are indexed at the start tag!)
104 		SpanQuery filter = new SpanQueryAnd(attrFilters);
105 		return new SpanQueryPositionFilter(allTags, filter, TextPatternPositionFilter.Operation.STARTS_AT, false);
106 	}
107 
108 	@Override
109 	public SpanQuery positionFilter(QueryExecutionContext context,
110 			SpanQuery producer, SpanQuery filter, Operation op, boolean invert,
111 			int leftAdjust, int rightAdjust) {
112 		return new SpanQueryPositionFilter(producer, filter, op, invert, leftAdjust, rightAdjust);
113 	}
114 
115 	@Override
116 	public SpanQuery startsAt(QueryExecutionContext context, SpanQuery producer, SpanQuery filter) {
117 		return new SpanQueryPositionFilter(producer, filter, TextPatternPositionFilter.Operation.STARTS_AT, false);
118 	}
119 
120 	@Override
121 	public SpanQuery endsAt(QueryExecutionContext context, SpanQuery producer, SpanQuery filter) {
122 		return new SpanQueryPositionFilter(producer, filter, TextPatternPositionFilter.Operation.ENDS_AT, false);
123 	}
124 
125 	@Override
126 	public SpanQuery term(QueryExecutionContext context, String value) {
127 		// Use a BlackLabSpanTermQuery instead of default Lucene one
128 		// because we need to override getField() to only return the base field name,
129 		// not the complete field name with the property.
130 		return new BLSpanTermQuery(new Term(context.luceneField(), context.optDesensitize(value)));
131 	}
132 
133 	@Override
134 	public SpanQuery expand(QueryExecutionContext context, SpanQuery clause, boolean expandToLeft, int min, int max) {
135 		SpanQueryExpansion spanQueryExpansion = new SpanQueryExpansion(clause, expandToLeft, min, max);
136 		spanQueryExpansion.setIgnoreLastToken(context.alwaysHasClosingToken());
137 		return spanQueryExpansion;
138 	}
139 
140 	@Override
141 	public SpanQuery filterNGrams(QueryExecutionContext context, SpanQuery clause, Operation op, int min, int max) {
142 		return new SpanQueryFilterNGrams(clause, op, min, max);
143 	}
144 
145 	@Override
146 	public SpanQuery repetition(SpanQuery clause, int min, int max) {
147 		return new SpanQueryRepetition(clause, min, max);
148 	}
149 
150 	@Override
151 	public SpanQuery docLevelAndNot(SpanQuery include, SpanQuery exclude) {
152 		return new SpanQueryDocLevelAndNot(include, exclude);
153 	}
154 
155 	@Override
156 	public SpanQuery wildcard(QueryExecutionContext context, String value) {
157 		return new BLSpanMultiTermQueryWrapper<WildcardQuery>(new WildcardQuery(new Term(context.luceneField(),
158 				context.optDesensitize(value))));
159 	}
160 
161 	@Override
162 	public SpanQuery prefix(QueryExecutionContext context, String value) {
163 		return new BLSpanMultiTermQueryWrapper<PrefixQuery>(new PrefixQuery(new Term(context.luceneField(),
164 				context.optDesensitize(value))));
165 	}
166 
167 	@Override
168 	public SpanQuery not(QueryExecutionContext context, SpanQuery clause) {
169 		SpanQueryNot spanQueryNot = new SpanQueryNot(clause);
170 		spanQueryNot.setIgnoreLastToken(context.alwaysHasClosingToken());
171 		return spanQueryNot;
172 	}
173 
174 	@Override
175 	public SpanQuery any(QueryExecutionContext context, int min, int max) {
176 		return new SpanQueryNGrams(context.alwaysHasClosingToken(), context.luceneField(), min, max);
177 		//return SpanQueryNot.matchAllTokens(context.alwaysHasClosingToken(), context.luceneField());
178 	}
179 
180 	@Override
181 	public SpanQuery edge(SpanQuery clause, boolean rightEdge) {
182 		return new SpanQueryEdge(clause, rightEdge);
183 	}
184 
185 	@Override
186 	public SpanQuery captureGroup(SpanQuery clause, String name) {
187 		return new SpanQueryCaptureGroup(clause, name);
188 	}
189 
190 }
