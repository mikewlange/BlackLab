16  package nl.inl.blacklab.search.lucene;
17  
18  import java.util.ArrayList;
19  import java.util.List;
20  import java.util.Map;
21  
22  import nl.inl.blacklab.index.complex.ComplexFieldUtil;
23  import nl.inl.blacklab.search.QueryExecutionContext;
24  import nl.inl.blacklab.search.TextPatternTranslator;
25  import nl.inl.blacklab.search.lucene.SpanQueryPositionFilter.Filter;
26  import nl.inl.blacklab.search.sequences.SpanQueryExpansion;
27  import nl.inl.blacklab.search.sequences.SpanQueryRepetition;
28  import nl.inl.blacklab.search.sequences.SpanQuerySequence;
29  
30  import org.apache.lucene.index.Term;
31  import org.apache.lucene.search.PrefixQuery;
32  import org.apache.lucene.search.RegexpQuery;
33  import org.apache.lucene.search.WildcardQuery;
34  import org.apache.lucene.search.spans.SpanQuery;
35  
36  /**
37   * Translates a TextPattern to a Lucene SpanQuery object.
38   */
39  public class TextPatternTranslatorSpanQuery extends TextPatternTranslator<SpanQuery> {
40  
41  	@Override
42  	public SpanQuery and(QueryExecutionContext context, List<SpanQuery> clauses) {
43  		return new SpanQueryAnd(clauses);
44  	}
45  
46  	@Override
47  	public SpanQuery or(QueryExecutionContext context, List<SpanQuery> clauses) {
48  		return new BLSpanOrQuery(clauses.toArray(new SpanQuery[] {}));
49  	}
50  
51  	@Override
52  	public SpanQuery regex(QueryExecutionContext context, String value) {
53  		String valueNoStartEndMatch = value.replaceAll("\\^|\\$", "");
54  		return new BLSpanMultiTermQueryWrapper<RegexpQuery>(new RegexpQuery(
55  				new Term(context.luceneField(), context.optDesensitize(valueNoStartEndMatch))));
56  	}
57  
58  	@Override
59  	public SpanQuery sequence(QueryExecutionContext context, List<SpanQuery> clauses) {
60  		return new SpanQuerySequence(clauses);
61  	}
62  
63  	@Override
64  	public SpanQuery docLevelAnd(QueryExecutionContext context, List<SpanQuery> clauses) {
65  		return new SpanQueryDocLevelAnd(clauses);
66  	}
67  
68  	@Override
69  	public SpanQuery fuzzy(QueryExecutionContext context, String value, int maxEdits, int prefixLength) {
70  		return new SpanFuzzyQuery(new Term(context.luceneField(), context.optDesensitize(value)), maxEdits, prefixLength);
71  	}
72  
73  	@Override
74  	public SpanQuery tags(QueryExecutionContext context, String elementName, Map<String, String> attr) {
75  		SpanQueryTags allTags = new SpanQueryTags(context, elementName);
76  		if (attr == null || attr.size() == 0)
77  			return allTags;
78  
79  		// Construct attribute filters
80  		List<SpanQuery> attrFilters = new ArrayList<SpanQuery>();
81  		QueryExecutionContext startTagContext = context.withProperty(ComplexFieldUtil.START_TAG_PROP_NAME);
82  		for (Map.Entry<String,String> e: attr.entrySet()) {
83  			String value = optInsensitive(context, "@" + e.getKey() + "__" + e.getValue());
84  			attrFilters.add(term(startTagContext, value));
85  		}
86  
87  		// Filter the tags
88  		// (NOTE: only works for start tags and full elements because attribute values
89  		//  are indexed at the start tag!)
90  		SpanQuery filter = new SpanQueryAnd(attrFilters);
91  		return new SpanQueryPositionFilter(allTags, filter, Filter.STARTS_AT);
92  	}
93  
94  	@Override
95  	public SpanQuery containing(QueryExecutionContext context, SpanQuery containers, SpanQuery search) {
96  		return new SpanQueryPositionFilter(containers, search, SpanQueryPositionFilter.Filter.CONTAINING);
97  	}
98  
99  	@Override
100 	public SpanQuery within(QueryExecutionContext context, SpanQuery search, SpanQuery containers) {
101 		return new SpanQueryPositionFilter(search, containers, SpanQueryPositionFilter.Filter.WITHIN);
102 	}
103 
104 	@Override
105 	public SpanQuery startsAt(QueryExecutionContext context, SpanQuery producer, SpanQuery filter) {
106 		return new SpanQueryPositionFilter(producer, filter, SpanQueryPositionFilter.Filter.STARTS_AT);
107 	}
108 
109 	@Override
110 	public SpanQuery endsAt(QueryExecutionContext context, SpanQuery producer, SpanQuery filter) {
111 		return new SpanQueryPositionFilter(producer, filter, SpanQueryPositionFilter.Filter.ENDS_AT);
112 	}
113 
114 	@Override
115 	public SpanQuery term(QueryExecutionContext context, String value) {
116 		// Use a BlackLabSpanTermQuery instead of default Lucene one
117 		// because we need to override getField() to only return the base field name,
118 		// not the complete field name with the property.
119 		return new BLSpanTermQuery(new Term(context.luceneField(), context.optDesensitize(value)));
120 	}
121 
122 	@Override
123 	public SpanQuery expand(QueryExecutionContext context, SpanQuery clause, boolean expandToLeft, int min, int max) {
124 		SpanQueryExpansion spanQueryExpansion = new SpanQueryExpansion(clause, expandToLeft, min, max);
125 		spanQueryExpansion.setIgnoreLastToken(context.alwaysHasClosingToken());
126 		return spanQueryExpansion;
127 	}
128 
129 	@Override
130 	public SpanQuery repetition(SpanQuery clause, int min, int max) {
131 		return new SpanQueryRepetition(clause, min, max);
132 	}
133 
134 	@Override
135 	public SpanQuery docLevelAndNot(SpanQuery include, SpanQuery exclude) {
136 		return new SpanQueryAndNot(include, exclude);
137 	}
138 
139 	@Override
140 	public SpanQuery wildcard(QueryExecutionContext context, String value) {
141 		return new BLSpanMultiTermQueryWrapper<WildcardQuery>(new WildcardQuery(new Term(context.luceneField(),
142 				context.optDesensitize(value))));
143 	}
144 
145 	@Override
146 	public SpanQuery prefix(QueryExecutionContext context, String value) {
147 		return new BLSpanMultiTermQueryWrapper<PrefixQuery>(new PrefixQuery(new Term(context.luceneField(),
148 				context.optDesensitize(value))));
149 	}
150 
151 	@Override
152 	public SpanQuery not(QueryExecutionContext context, SpanQuery clause) {
153 		SpanQueryNot spanQueryNot = new SpanQueryNot(clause);
154 		spanQueryNot.setIgnoreLastToken(context.alwaysHasClosingToken());
155 		return spanQueryNot;
156 	}
157 
158 	@Override
159 	public SpanQuery any(QueryExecutionContext context) {
160 		return SpanQueryNot.matchAllTokens(context.alwaysHasClosingToken(), context.luceneField());
161 	}
162 
163 	@Override
164 	public SpanQuery edge(SpanQuery clause, boolean rightEdge) {
165 		return new SpanQueryEdge(clause, rightEdge);
166 	}
167 
168 	@Override
169 	public SpanQuery captureGroup(SpanQuery clause, String name) {
170 		return new SpanQueryCaptureGroup(clause, name);
171 	}
172 
173 }
