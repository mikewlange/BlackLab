35  /**
36   * Expands the source spans to the left and right by the given ranges.
37   *
38   * This is used to support sequences including subsequences of completely unknown tokens (like
39   * "apple" []{2, 4} "pear" to find apple and pear with 2 to 4 tokens in between).
40   *
41   * Note that this class will generate all possible expansions, so if you call it with left-expansion
42   * of between 2 to 4 tokens, it will generate 3 new hits for every hit from the source spans: one
43   * hit with 2 more tokens to the left, one hit with 3 more tokens to the left, and one hit with 4
44   * more tokens to the left.
45   *
46   * Spans generated from this query will be sorted by start point and then by end point, and any
47   * duplicates generated will be discarded.
48   */
49  public class SpanQueryExpansion extends BLSpanQueryAbstract {
50  
51  	/** Whether to expand to left (true) or right (false) */
52  	boolean expandToLeft;
53  
54  	/** Minimum number of tokens to expand */
55  	int min;
56  
57  	/** Maximum number of tokens to expand (MAX_UNLIMITED = infinite) */
58  	int max;
59  
60  	/** if true, we assume the last token is always a special closing token and ignore it */
61  	boolean ignoreLastToken = false;
62  
63  	public boolean isIgnoreLastToken() {
64  		return ignoreLastToken;
65  	}
66  
67  	public SpanQueryExpansion(BLSpanQuery clause, boolean expandToLeft, int min, int max) {
68  		super(clause);
69  		this.expandToLeft = expandToLeft;
70  		this.min = min;
71  		this.max = max == -1 ? MAX_UNLIMITED : max;
72  		if (min > this.max)
73  			throw new IllegalArgumentException("min > max");
74  		if (min < 0 || this.max < 0)
75  			throw new IllegalArgumentException("Expansions cannot be negative");
76  	}
77  
78  	@Override
79  	public boolean equals(Object o) {
80  		if (this == o)
81  			return true;
82  		if (!super.equals(o))
83  			return false;
84  
85  		final SpanQueryExpansion that = (SpanQueryExpansion) o;
86  		return expandToLeft == that.expandToLeft && min == that.min && max == that.max;
87  	}
88  
89  	@Override
90  	public BLSpanQuery rewrite(IndexReader reader) throws IOException {
91  		List<BLSpanQuery> rewritten = rewriteClauses(reader);
92  		if (rewritten == null)
93  			return this;
94  		if (min == 0 && max == 0)
95  			return rewritten.get(0); // not really an expansion
96  		SpanQueryExpansion result = new SpanQueryExpansion(rewritten.get(0), expandToLeft, min, max);
97  		if (ignoreLastToken)
98  			result.setIgnoreLastToken(true);
99  		return result;
100 	}
101 
102 	@Override
103 	public boolean matchesEmptySequence() {
104 		return clauses.get(0).matchesEmptySequence() && min == 0;
105 	}
106 
107 	@Override
108 	public BLSpanQuery noEmpty() {
109 		if (!matchesEmptySequence())
110 			return this;
111 		int newMin = min == 0 ? 1 : min;
112 		SpanQueryExpansion result = new SpanQueryExpansion(clauses.get(0).noEmpty(), expandToLeft, newMin, max);
113 		result.setIgnoreLastToken(ignoreLastToken);
114 		return result;
115 	}
116 
117 	@Override
118 	public BLSpanWeight createWeight(IndexSearcher searcher, boolean needsScores) throws IOException {
119 		BLSpanWeight weight = clauses.get(0).createWeight(searcher, needsScores);
120 		return new SpanWeightExpansion(weight, searcher, needsScores ? getTermContexts(weight) : null);
121 	}

123 	class SpanWeightExpansion extends BLSpanWeight {
124 
125 		final BLSpanWeight weight;
126 
127 		public SpanWeightExpansion(BLSpanWeight weight, IndexSearcher searcher, Map<Term, TermContext> terms) throws IOException {
128 			super(SpanQueryExpansion.this, searcher, terms);
129 			this.weight = weight;
130 		}
131 
132 		@Override
133 		public void extractTerms(Set<Term> terms) {
134 			weight.extractTerms(terms);
135 		}
136 
137 		@Override
138 		public void extractTermContexts(Map<Term, TermContext> contexts) {
139 			weight.extractTermContexts(contexts);
140 		}
141 
142 		@Override
143 		public BLSpans getSpans(final LeafReaderContext context, Postings requiredPostings) throws IOException {
144 			BLSpans spansSource = weight.getSpans(context, requiredPostings);
145 			if (spansSource == null)
146 				return null;
147 			BLSpans spans = new SpansExpansionRaw(ignoreLastToken, context.reader(), clauses.get(0).getField(), spansSource, expandToLeft, min, max);
148 			return spans;
149 		}

153 	@Override
154 	public int hashCode() {
155 		int h = clauses.hashCode();
156 		h ^= (h << 10) | (h >>> 23);
157 		h ^= min << 10;
158 		h ^= max << 5;
159 		h ^= expandToLeft ? 1 : 0;
160 		return h;
161 	}

164 	public String toString(String field) {
165 		return "EXPAND(" + clauses.get(0) + ", " + (expandToLeft ? "L" : "R") + ", " + min + ", " + inf(max) + ")";
166 	}

232 	@Override
233 	public Nfa getNfa(ForwardIndexAccessor fiAccessor, int direction) {
234 		if (max == MAX_UNLIMITED)
235 			throw new UnsupportedOperationException("Unlimited expansion using forward index not implemented");
236 		Nfa nfa = clauses.get(0).getNfa(fiAccessor, direction);
237 		NfaState any = new NfaStateAnyToken(clauses.get(0).getRealField(), null);
238 		Nfa frag = new Nfa(any, Arrays.asList(any));
239 		frag.repeat(min, max);
240 		if (expandToLeft && direction == 1 || !expandToLeft && direction == -1) {
241 			// Prepend nfa with stretch of anytokens
242 			frag.append(nfa);
243 			nfa = frag;
244 		} else {
245 			// Append stretch of anytokens to nfa
246 			nfa.append(frag);
247 		}
248 		return nfa;
249 	}
250 
251 	@Override
252 	public boolean canMakeNfa() {
253 		return max != MAX_UNLIMITED && clauses.get(0).canMakeNfa();
254 	}

256 	@Override
257 	public long reverseMatchingCost(IndexReader reader) {
258 		int numberOfExpansionSteps = max == MAX_UNLIMITED ? 50 : max - min + 1;
259 		return clauses.get(0).reverseMatchingCost(reader) * numberOfExpansionSteps;
260 	}
261 
262 	@Override
263 	public int forwardMatchingCost() {
264 		int cost = clauses.get(0).forwardMatchingCost();
265 		int nMax = max == MAX_UNLIMITED ? 50 : max;
266 		for (int i = min; i <= nMax; i++) {
267 			cost += i;
268 		}
269 		return cost;
270 	}
271 
272 	public BLSpanQuery addExpand(int addMin, int addMax) {
273 		int nMin = min + addMin;
274 		int nMax = addMaxValues(max, addMax);
275 		SpanQueryExpansion result = new SpanQueryExpansion(clauses.get(0), expandToLeft, nMin, nMax);
276 		result.setIgnoreLastToken(isIgnoreLastToken());
277 		return result;
278 	}
279 	
280 	@Override
281 	public boolean canInternalizeNeighbour(BLSpanQuery clause, boolean onTheRight) {
282 		if (onTheRight == expandToLeft) {
283 			// Internalization on the side of our non-expanded clause. Always possible.
284 			return true;
285 		}
286 		// Internalization on the side of our expansion. Only possible for any token clauses.
287 		return clause instanceof SpanQueryAnyToken;
288 	}
289 
290 	@Override
291 	public BLSpanQuery internalizeNeighbour(BLSpanQuery clause, boolean onTheRight) {
292 		if (onTheRight != expandToLeft && !(clause instanceof SpanQueryAnyToken)) {
293 			throw new IllegalArgumentException("Cannot internalize " + clause + " into " + this + " on the " + (onTheRight ? "right" : "left") + "side");
294 		}
295 		if (onTheRight == expandToLeft) {
296 			// "Gobble up" a clause into the clause we're expanding.
297 			// If we're expanding to the left, the clause is added to the right of what we were expanding, and vice versa.
298 			SpanQuerySequence seq = SpanQuerySequence.sequenceInternalize(clauses.get(0), clause, expandToLeft);
299 			SpanQueryExpansion result = new SpanQueryExpansion(seq, expandToLeft, min, max);
300 			result.setIgnoreLastToken(ignoreLastToken);
301 			return result;
302 		}
303 		// Add any token to our expansion.
304 		return addExpand(clause.hitsLengthMin(), clause.hitsLengthMax());
305 	}
