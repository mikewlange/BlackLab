28  /**
29   * Filters hits from a producer query based on the hit positions of a filter query.
30   * This allows us to do several things, such as:
31   * * find hits from the producer that contain one or more hits from the filter
32   * * find hits from the producer are contained by hit(s) from the filter
33   * * find hits from the producer that start at the same position as a hit from the filter
34   * * find hits from the producer that end at the same position as a hit from the filter
35   */
36  public class SpanQueryPositionFilter extends BLSpanQueryAbstract {
37  
38  	static SpanComparatorStartPoint cmpStartPoint = new SpanComparatorStartPoint();
39  
40  	/** Filter operation to apply */
41  	SpanQueryPositionFilter.Operation op;
42  
43  	/** Return producer spans that DON'T match the filter instead? */
44  	boolean invert;
45  
46  	/** How to adjust the left edge of the producer hits while matching */
47  	int leftAdjust;
48  
49  	/** How to adjust the right edge of the producer hits while matching */
50  	int rightAdjust;
51  
52  	/**
53  	 * Produce hits that match filter hits.
54  	 *
55  	 * @param producer hits we may be interested in
56  	 * @param filter how we determine what producer hits we're interested in
57  	 * @param op operation used to determine what producer hits we're interested in (containing, within, startsat, endsat)
58  	 * @param invert produce hits that don't match filter instead?
59  	 */
60  	public SpanQueryPositionFilter(BLSpanQuery producer, BLSpanQuery filter, SpanQueryPositionFilter.Operation op, boolean invert) {
61  		this(producer, filter, op, invert, 0, 0);
62  	}
63  
64  	/**
65  	 * Produce hits that match filter hits.
66  	 *
67  	 * @param producer hits we may be interested in
68  	 * @param filter how we determine what producer hits we're interested in
69  	 * @param op operation used to determine what producer hits we're interested in (containing, within, startsat, endsat)
70  	 * @param invert produce hits that don't match filter instead?
71  	 * @param leftAdjust how to adjust the left edge of the producer hits while matching
72  	 * @param rightAdjust how to adjust the right edge of the producer hits while matching
73  	 */
74  	public SpanQueryPositionFilter(BLSpanQuery producer, BLSpanQuery filter, SpanQueryPositionFilter.Operation op, boolean invert, int leftAdjust, int rightAdjust) {
75  		super(producer, filter);
76  		this.op = op;
77  		this.invert = invert;
78  		this.leftAdjust = leftAdjust;
79  		this.rightAdjust = rightAdjust;
80  	}
81  
82  	@Override
83  	public BLSpanQuery rewrite(IndexReader reader) throws IOException {
84  		BLSpanQuery producer = clauses.get(0).rewrite(reader);
85  		BLSpanQuery filter = clauses.get(1).rewrite(reader);
86  
87  		if (!invert && op != SpanQueryPositionFilter.Operation.STARTS_AT && op != SpanQueryPositionFilter.Operation.ENDS_AT && producer instanceof SpanQueryAnyToken) {
88  			// We're filtering "all n-grams of length min-max".
89  			// Use the special optimized SpanQueryFilterNGrams.
90  			SpanQueryAnyToken tp = (SpanQueryAnyToken)producer;
91  			return new SpanQueryFilterNGrams(filter, op, tp.hitsLengthMin(), tp.hitsLengthMax());
92  		}
93  
94  		if (producer != clauses.get(0) || filter != clauses.get(1)) {
95  			SpanQueryPositionFilter result = new SpanQueryPositionFilter(producer, filter, op, invert);
96  			result.leftAdjust = leftAdjust;
97  			result.rightAdjust = rightAdjust;
98  			return result;
99  		}
100 		return this;
101 	}
102 
103 	@Override
104 	public BLSpanWeight createWeight(IndexSearcher searcher, boolean needsScores) throws IOException {
105 		BLSpanWeight prodWeight = clauses.get(0).createWeight(searcher, needsScores);
106 		BLSpanWeight filterWeight = clauses.get(1).createWeight(searcher, needsScores);
107 		Map<Term, TermContext> contexts = needsScores ? getTermContexts(prodWeight, filterWeight) : null;
108 		return new SpanWeightPositionFilter(prodWeight, filterWeight, searcher, contexts);
109 	}

111 	class SpanWeightPositionFilter extends BLSpanWeight {
112 
113 		final BLSpanWeight prodWeight, filterWeight;
114 
115 		public SpanWeightPositionFilter(BLSpanWeight prodWeight, BLSpanWeight filterWeight, IndexSearcher searcher, Map<Term, TermContext> terms) throws IOException {
116 			super(SpanQueryPositionFilter.this, searcher, terms);
117 			this.prodWeight = prodWeight;
118 			this.filterWeight = filterWeight;
119 		}
120 
121 		@Override
122 		public void extractTerms(Set<Term> terms) {
123 			prodWeight.extractTerms(terms);
124 			filterWeight.extractTerms(terms);
125 		}
126 
127 		@Override
128 		public void extractTermContexts(Map<Term, TermContext> contexts) {
129 			prodWeight.extractTermContexts(contexts);
130 			filterWeight.extractTermContexts(contexts);
131 		}
132 
133 		@Override
134 		public BLSpans getSpans(final LeafReaderContext context, Postings requiredPostings) throws IOException {
135 			BLSpans spansProd = prodWeight.getSpans(context, requiredPostings);
136 			if (spansProd == null)
137 				return null;
138 			if (!clauses.get(0).hitsStartPointSorted())
139 				spansProd = new PerDocumentSortedSpans(spansProd, PerDocumentSortedSpans.cmpStartPoint, false);
140 			BLSpans spansFilter = filterWeight.getSpans(context, requiredPostings);
141 			if (spansFilter == null) {
142 				// No filter hits. If it's a positive filter, that means no producer hits can match.
143 				// If it's a negative filter, all producer hits match.
144 				return invert ? spansProd : null;
145 			}
146 			boolean filterFixedLength = clauses.get(1).hitsAllSameLength();
147 			SpansInBuckets filter;
148 			if (clauses.get(1).hitsStartPointSorted()) {
149 				// Already start point sorted; no need to sort buckets again
150 				filter = new SpansInBucketsPerDocument(spansFilter);
151 			} else {
152 				// Not sorted yet; sort buckets
153 				filter = new SpansInBucketsPerDocumentSorted(spansFilter, cmpStartPoint);
154 			}
155 			return new SpansPositionFilter(spansProd, filter, filterFixedLength, op, invert, leftAdjust, rightAdjust);
156 		}
157 	}
158 
159 	/** The different positional operations */
160 	public enum Operation {
161 
162 		/** Producer hit contains filter hit */
163 		CONTAINING,
164 
165 		/** Producer hit contained in filter hit */
166 		WITHIN,
167 
168 		/** Producer hit starts at filter hit */
169 		STARTS_AT,
170 
171 		/** Producer hit ends at filter hit */
172 		ENDS_AT,
173 
174 		/** Producer hit exactly matches filter hit */
175 		MATCHES,
176 
177 		/** Producer hit contains filter hit, at its end */
178 		CONTAINING_AT_START,
179 
180 		/** Producer hit contains filter hit, at its start*/
181 		CONTAINING_AT_END
182 
183 	}

185 	@Override
186 	public String toString(String field) {
187 		String not = invert ? "NOT" : "";
188 		String adj = (leftAdjust != 0 || rightAdjust != 0 ? ", " + leftAdjust + ", " + rightAdjust : "");
189 		return "POSFILTER(" + clausesToString(field) + ", " + not + op + adj + ")";
190 	}

197 	 * Adjust the left edge of the producer hits for matching only.
198 	 *
199 	 * That is, the original producer hit is returned, not the adjusted one.
200 	 *
201 	 * @param delta how to adjust the edge
202 	 */
203 	public void adjustLeft(int delta) {
204 		leftAdjust += delta;
205 	}

207 	/**
208 	 * Adjust the right edge of the producer hits for matching only.

213 	 */
214 	public void adjustRight(int delta) {
215 		rightAdjust += delta;
216 	}

283 	@Override
284 	public BLSpanQuery internalizeNeighbour(BLSpanQuery clause, boolean addToRight) {
285 		if (!clause.hitsAllSameLength())
286 			throw new RuntimeException("Trying to internalize non-constant-length clause: " + clause);
287 		// Create a new position filter query with a constant-length clause added to our producer.
288 		// leftAdjust and rightAdjust are updated according to the clause's length, so it is not
289 		// actually filtered.
290 		BLSpanQuery producer = clauses.get(0);
291 		SpanQuerySequence seq = SpanQuerySequence.sequenceInternalize(producer, clause, addToRight);
292 		if (addToRight)
293 			return new SpanQueryPositionFilter(seq, clauses.get(1), op, invert, leftAdjust, rightAdjust - clause.hitsLengthMin());
294 		return new SpanQueryPositionFilter(seq, clauses.get(1), op, invert, leftAdjust + clause.hitsLengthMin(), rightAdjust);
295 	}
