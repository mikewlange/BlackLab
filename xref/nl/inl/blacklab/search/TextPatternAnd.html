16  package nl.inl.blacklab.search;

18  import java.util.ArrayList;
19  import java.util.List;
20  
21  /**
22   * AND query for combining different properties from a complex field.
23   *
24   * Note that when generating a SpanQuery, the Span start and end are also matched! Therefore only
25   * two hits in the same document at the same start and end position will produce a match. This is
26   * useful for e.g. selecting adjectives that start with a 'b' (queries on different property
27   * (sub)fields that should apply to the same word).
28   */
29  public class TextPatternAnd extends TextPatternCombiner {
30  	public TextPatternAnd(TextPattern... clauses) {
31  		super(clauses);
32  	}
33  
34  	@Override
35  	public <T> T translate(TextPatternTranslator<T> translator, QueryExecutionContext context) {
36  		List<T> chResults = new ArrayList<T>(clauses.size());
37  		for (TextPattern cl : clauses) {
38  			chResults.add(cl.translate(translator, context));
39  		}
40  		if (chResults.size() == 1)
41  			return chResults.get(0);
42  		return translator.and(context, chResults);
43  	}
44  
45  	/*
46  
47  	NOTE: this code rewrites AND queries containing only NOT children
48  	into OR queries. It is not currently used but we may want to use something
49  	like this in the future to better optimize certain queries, so we'll leave
50  	it here for now.
51  
52  	@Override
53  	public TextPattern rewrite() {
54  		boolean hasOnlyNotChildren = true;
55  		for (TextPattern child: clauses) {
56  			if (!(child instanceof TextPatternNot)) {
57  				hasOnlyNotChildren = false;
58  				break;
59  			}
60  		}
61  		if (hasOnlyNotChildren) {
62  			// Node should be rewritten to OR
63  			TextPattern[] rewrittenAndInv = new TextPattern[clauses.size()];
64  			for (int i = 0; i < clauses.size(); i++) {
65  				rewrittenAndInv[i] = clauses.get(i).rewrite().inverted();
66  			}
67  			return new TextPatternNot(new TextPatternOr(rewrittenAndInv));
68  		}
69  		// Node need not be rewritten
70  		return this;
71  	}
