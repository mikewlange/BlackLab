1   package nl.inl.blacklab.search.grouping;
2   
3   import nl.inl.blacklab.forwardindex.Terms;
4   import nl.inl.blacklab.index.complex.ComplexFieldUtil;
5   import nl.inl.blacklab.search.Hits;
6   
7   public class HitPropValueContextWord extends HitPropValueContext {
8   	int valueTokenId;
9   
10  	int valueSortOrder;
11  
12  	boolean sensitive;
13  
14  	public HitPropValueContextWord(Hits hits, String propName, int value, boolean sensitive) {
15  		super(hits, propName);
16  		this.valueTokenId = value;
17  		this.sensitive = sensitive;
18  		valueSortOrder = value < 0 ? value : terms.idToSortPosition(value, sensitive);
19  	}
20  
21  	@Override
22  	public int compareTo(Object o) {
23  		int a = valueSortOrder, b = ((HitPropValueContextWord)o).valueSortOrder;
24  		return a == b ? 0 : (a < b ? -1 : 1);
25  	}
26  
27  	@Override
28  	public int hashCode() {
29  		return ((Integer)valueSortOrder).hashCode();
30  	}
31  
32  	@Override
33  	public String toString() {
34  		return valueTokenId < 0 ? "-" : terms.get(valueTokenId);
35  	}
36  
37  	public static HitPropValue deserialize(Hits hits, String info) {
38  		String[] parts = PropValSerializeUtil.splitParts(info);
39  		String fieldName = hits.settings().concordanceField();
40  		String propName = parts[0];
41  		boolean sensitive = parts[1].equalsIgnoreCase("s");
42  		Terms termsObj = hits.getSearcher().getForwardIndex(ComplexFieldUtil.propertyField(fieldName, propName)).getTerms();
43  		int id;
44  		if (parts[2].length() == 0)
45  			id = Terms.NO_TERM; // no token
46  		else
47  			id = termsObj.indexOf(parts[2]);
48  		return new HitPropValueContextWord(hits, propName, id, sensitive);
49  	}
50  
51  	@Override
52  	public String serialize() {
53  		String token;
54  		if (valueTokenId < 0)
55  			token = ""; // no token
56  		else
57  			token = terms.get(valueTokenId);
58  		return PropValSerializeUtil.combineParts(
59  			"cwo", propName,
60  			(sensitive ? "s" : "i"),
61  			token);
62  	}
63  }
