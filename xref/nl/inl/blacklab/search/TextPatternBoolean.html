16  package nl.inl.blacklab.search;
17  
18  import java.util.ArrayList;
19  import java.util.Arrays;
20  import java.util.List;
21  
22  import org.apache.lucene.search.BooleanClause.Occur;
23  
24  /**
25   * A TextPattern that performs document-level combining of boolean clauses. Modeled after Lucene's
26   * BooleanQuery, each clause can get the MUST, SHOULD or MUST NOT label attached.
27   *
28   * MUST-clauses must all occur in the document. At least one of all SHOULD clauses must occur in the
29   * document. None of the MUST NOT clauses must occur in the document.
30   *
31   * For documents that satisfy these criteria, all the MUST and SHOULD hits are reported.
32   * @deprecated not used
33   */
34  @Deprecated
35  public class TextPatternBoolean extends TextPattern {
36  
37  	private List<TextPattern> must = new ArrayList<>();
38  
39  	private List<TextPattern> should = new ArrayList<>();
40  
41  	private List<TextPattern> mustNot = new ArrayList<>();
42  
43  	public TextPatternBoolean(boolean disableCoord) {
44  		// do nothing (disableCoord is ignored as we don't use scoring)
45  	}
46  
47  	@Override
48  	public <T> T translate(TextPatternTranslator<T> translator, QueryExecutionContext context) {
49  
50  		// First, translate clauses into complete TextPattern. Then translate that.
51  		TextPattern translated = null;
52  
53  		// MUST and SHOULD
54  		TextPattern tpMust = null, tpShould = null;
55  		if (must.size() > 0) {
56  			// Build a TextPattern that combines all MUST queries with AND
57  			if (must.size() == 1)
58  				tpMust = must.get(0);
59  			else
60  				tpMust = new TextPatternDocLevelAnd(must.toArray(new TextPattern[0]));
61  		}
62  		if (should.size() > 0) {
63  			if (should.size() == 1)
64  				tpShould = should.get(0);
65  			else {
66  				// Build a TextPattern that combines all SHOULD queries with OR
67  				tpShould = new TextPatternOr(should.toArray(new TextPattern[0]));
68  			}
69  		}
70  		if (tpMust == null && tpShould == null)
71  			throw new RuntimeException("Query must contain included terms (cannot just exclude)");
72  		else if (tpMust != null && tpShould != null) {
73  			// Require all MUST queries and at least one of the SHOULD queries
74  			translated = new TextPatternDocLevelAnd(tpMust, tpShould);
75  		} else {
76  			translated = tpMust == null ? tpShould : tpMust;
77  		}
78  
79  		// MUST NOT
80  		if (mustNot.size() > 0) {
81  			// Build a TextPattern that combines all MUST NOT queries with AND
82  			translated = new TextPatternDocLevelAndNot(translated, new TextPatternDocLevelAnd(
83  					mustNot.toArray(new TextPattern[0])));
84  		}
85  
86  		return translated.translate(translator, context);
87  	}
88  
89  	public void add(TextPattern query, Occur occur) {
90  		add(new TPBooleanClause(query, occur));
91  	}
92  
93  	public void add(TPBooleanClause clause) {
94  		switch (clause.getOccur()) {
95  		case FILTER: // same as MUST, but does not participate in scoring
96  		case MUST:
97  			must.add(clause.getQuery());
98  			break;
99  		case SHOULD:
100 			should.add(clause.getQuery());
101 			break;
102 		case MUST_NOT:
103 			mustNot.add(clause.getQuery());
104 			break;
105 		}
106 	}
107 
108 	@Override
109 	public Object clone() {
110 		try {
111 			TextPatternBoolean clone = (TextPatternBoolean) super.clone();
112 
113 			// copy list of children so we can modify it independently
114 			clone.must = new ArrayList<>(must);
115 			clone.should = new ArrayList<>(should);
116 			clone.mustNot = new ArrayList<>(mustNot);
117 
118 			return clone;
119 		} catch (CloneNotSupportedException e) {
120 			throw new RuntimeException("Clone not supported: " + e.getMessage());
121 		}
122 	}
123 
124 	@Override
125 	public boolean equals(Object obj) {
126 		if (obj instanceof TextPatternBoolean) {
127 			TextPatternBoolean tp = ((TextPatternBoolean) obj);
128 			return must.equals(tp.must) && should.equals(tp.should) && mustNot.equals(tp.mustNot);
129 		}
130 		return false;
131 	}
132 
133 	@Override
134 	public boolean hasConstantLength() {
135 		int l = must.get(0).getMinLength();
136 		for (List<TextPattern> coll: Arrays.asList(must, should)) {
137 			for (TextPattern clause: coll) {
138 				if (!clause.hasConstantLength() || clause.getMinLength() != l)
139 					return false;
140 			}
141 		}
142 		return true;
143 	}
144 
145 	@Override
146 	public int getMinLength() {
147 		int n = Integer.MAX_VALUE;
148 		for (List<TextPattern> coll: Arrays.asList(must, should)) {
149 			for (TextPattern clause: coll) {
150 				n = Math.min(n, clause.getMinLength());
151 			}
152 		}
153 		return n;
154 	}
155 
156 	@Override
157 	public int getMaxLength() {
158 		int n = 0;
159 		for (List<TextPattern> coll: Arrays.asList(must, should)) {
160 			for (TextPattern clause: coll) {
161 				int l = clause.getMaxLength();
162 				if (l < 0)
163 					return -1; // infinite
164 				n = Math.max(n, l);
165 			}
166 		}
167 		return n;
168 	}

170 	@Override
171 	public TextPattern rewrite() {
172 
173 		// FIXME: rewrite clauses!
174 
175 		return this;
176 	}
177 
178 	@Override
179 	public int hashCode() {
180 		return must.hashCode() + should.hashCode() + mustNot.hashCode();
181 	}
182 
183 	@Override
184 	public String toString(QueryExecutionContext context) {
185 		return "BOOLEAN(" + clausesToString(must, context) + ", " + clausesToString(should, context) + ", " + clausesToString(mustNot, context) + ")";
186 	}
