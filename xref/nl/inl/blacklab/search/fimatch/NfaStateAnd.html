15  		this.nextStates = new ArrayList<>(nextStates);

22  	@Override
23  	public boolean findMatchesInternal(ForwardIndexDocument fiDoc, int pos, int direction, Set<Integer> matchEnds) {
24  		// Split state. Find matches for all alternatives.
25  		Set<Integer> newHitsFound = null;
26  		for (NfaState nextState: nextStates) {
27  			Set<Integer> matchesForClause = new HashSet<>();
28  			if (!nextState.findMatchesInternal(fiDoc, pos, direction, matchesForClause))
29  				return false; // short-circuit
30  			if (newHitsFound == null) {
31  				newHitsFound = matchesForClause;
32  			} else {
33  				// Calculate intersection
34  				newHitsFound.retainAll(matchesForClause);
35  				if (newHitsFound.size() == 0)
36  					return false; // no hits left; short-circuit
37  			}
38  		}
39  		if (matchEnds != null)
40  			matchEnds.addAll(newHitsFound);
41  		return true;
42  	}

44  	@Override
45  	void fillDangling(NfaState state) {
46  		for (int i = 0; i < nextStates.size(); i++) {
47  			if (nextStates.get(i) == null)
48  				nextStates.set(i, state);
49  		}
50  	}

52  	@Override
53  	NfaStateAnd copyInternal(Collection<NfaState> dangling, Map<NfaState, NfaState> copiesMade) {
54  		List<NfaState> clauseCopies = new ArrayList<>();
55  		boolean hasNulls = false;
56  		NfaStateAnd copy = new NfaStateAnd();
57  		copiesMade.put(this, copy);
58  		for (NfaState nextState: nextStates) {
59  			if (nextState == null)
60  				hasNulls = true;
61  			else
62  				nextState = nextState.copy(dangling, copiesMade);
63  			clauseCopies.add(nextState);
64  		}
65  		copy.nextStates.addAll(clauseCopies);
66  		if (hasNulls && dangling != null)
67  			dangling.add(copy);
68  		return copy;
69  	}

76  	@Override
77  	public boolean matchesEmptySequence(Set<NfaState> statesVisited) {
78  		if (statesVisited.contains(this)) {
79  			// We've found a cycle. Stop processing, and just return the
80  			// "safest" (least-guarantee) answer. In this case: we can't
81  			// guarantee that this DOESN'T match the empty sequence.
82  			return true;
83  		}
84  		statesVisited.add(this);
85  		for (NfaState nextState: nextStates) {
86  			if (!nextState.matchesEmptySequence(statesVisited))
87  				return false;
88  		}
89  		return true;
90  	}

92  	@Override
93  	public boolean hitsAllSameLength(Set<NfaState> statesVisited) {
94  		if (statesVisited.contains(this)) {
95  			// We've found a cycle. Stop processing, and just return the
96  			// "safest" (least-guarantee) answer. In this case: we can't
97  			// guarantee that hits are all the same length.
98  			return false;
99  		}
100 		statesVisited.add(this);
101 		for (NfaState nextState: nextStates) {
102 			// If any of the following states has all-same-length hits,
103 			// this state must also have them.
104 			if (nextState.hitsAllSameLength(statesVisited))
105 				return true;
106 		}
107 		return true;
108 	}

110 	@Override
111 	public int hitsLengthMin(Set<NfaState> statesVisited) {
112 		int hitLengthMin = Integer.MAX_VALUE;
113 		if (statesVisited.contains(this)) {
114 			// We've found a cycle. Stop processing, and just return the
115 			// "safest" (least-guarantee) answer. In this case: the smallest
116 			// hit might be 0 long.
117 			return 0;
118 		}
119 		statesVisited.add(this);
120 		for (NfaState nextState: nextStates) {
121 			int i = nextState.hitsLengthMin(statesVisited);
122 			if (i < hitLengthMin)
123 				hitLengthMin = i;
124 		}
125 		return hitLengthMin;
126 	}

128 	@Override
129 	public int hitsLengthMax(Set<NfaState> statesVisited) {
130 		int hitLengthMax = 0;
131 		if (statesVisited.contains(this)) {
132 			// We've found a cycle. Stop processing, and just return the
133 			// "safest" (least-guarantee) answer. In this case: the largest
134 			// hit might be "infinitely" large.
135 			return Integer.MAX_VALUE;
136 		}
137 		statesVisited.add(this);
138 		for (NfaState nextState: nextStates) {
139 			int i = nextState.hitsLengthMax(statesVisited);
140 			if (i > hitLengthMax)
141 				hitLengthMax = i;
142 		}
143 		return hitLengthMax;
144 	}

146 	@Override
147 	protected String dumpInternal(Map<NfaState, Integer> stateNrs) {
148 		StringBuilder b = new StringBuilder();
149 		for (NfaState s: nextStates) {
150 			if (b.length() > 0)
151 				b.append(",");
152 			b.append(dump(s, stateNrs));
153 		}
154 		return "AND(" + b.toString() + ")";
155 	}

157 	@Override
158 	void lookupPropertyNumbersInternal(ForwardIndexAccessor fiAccessor, Map<NfaState, Boolean> statesVisited) {
159 		for (NfaState s: nextStates) {
160 			if (s != null)
161 				s.lookupPropertyNumbers(fiAccessor, statesVisited);
162 		}
163 	}

165 	@Override
166 	protected void finishInternal(Set<NfaState> visited) {
167 		for (int i = 0; i < nextStates.size(); i++) {
168 			NfaState s = nextStates.get(i);
169 			if (s == null)
170 				nextStates.set(i, match());
171 			else
172 				s.finish(visited);
173 		}
174 	}
