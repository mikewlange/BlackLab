10  /**
11   * An OR-node where none of the clauses cycle back to an
12   * earlier node. This can be evaluated more efficiently,
13   * by finding matches for each of the clauses, and then
14   * continuing at nextState from each of the matches found.
15   */
16  public class NfaStateOrAcyclic extends NfaState {
17  
18  	List<NfaState> clauses;
19  
20  	NfaState nextState;
21  
22  	/** If we know all hits are same length (in finish()), we can optimize matching by shortcircuiting OR */
23  	boolean clausesAllSameLength;
24  
25  	public NfaStateOrAcyclic(List<NfaState> clauses, boolean clausesAllSameLength) {
26  		for (NfaState clause: clauses) {
27  			clause.finish(new HashSet<NfaState>());
28  		}
29  		this.clauses = new ArrayList<>(clauses);
30  		this.nextState = null;
31  		this.clausesAllSameLength = clausesAllSameLength;
32  	}
33  
34  	public NfaStateOrAcyclic() {
35  		this.clauses = new ArrayList<>();
36  		this.nextState = null;
37  	}

39  	@Override
40  	public boolean findMatchesInternal(ForwardIndexDocument fiDoc, int pos, int direction, Set<Integer> matchEnds) {
41  		// OR/Split state. Find matches for all alternatives.
42  		boolean clauseMatched = false;
43  		Set<Integer> clauseMatchEnds = new HashSet<>();
44  		if (clausesAllSameLength) {
45  			// We can short-circuit as soon as we find a single clause hit, because there can only be one match end.
46  			for (NfaState clause: clauses) {
47  				boolean matchesFound = false;
48  				matchesFound = clause.findMatchesInternal(fiDoc, pos, direction, clauseMatchEnds);
49  				clauseMatched |= matchesFound;
50  				if (clauseMatched)
51  					break; // short-circuit OR
52  			}
53  		} else {
54  			// We have to process all clauses because we need all match ends for the next phase.
55  			for (NfaState clause: clauses) {
56  				boolean matchesFound = false;
57  				matchesFound = clause.findMatchesInternal(fiDoc, pos, direction, clauseMatchEnds);
58  				clauseMatched |= matchesFound;
59  			}
60  		}
61  		boolean foundMatch = false;
62  		if (clauseMatched) {
63  			// Continue matching from the matches to our OR clauses
64  			for (Integer clauseMatchEnd: clauseMatchEnds) {
65  				foundMatch |= nextState.findMatchesInternal(fiDoc, clauseMatchEnd, direction, matchEnds);
66  				if (foundMatch && matchEnds == null)
67  					break; // we don't care about the match ends, just that there are matches
68  			}
69  		}
70  		return foundMatch;
71  	}

73  	@Override
74  	void fillDangling(NfaState state) {
75  		if (nextState == null)
76  			nextState = state;
77  	}
78  
79  	@Override
80  	NfaStateOrAcyclic copyInternal(Collection<NfaState> dangling, Map<NfaState, NfaState> copiesMade) {
81  		NfaStateOrAcyclic copy = new NfaStateOrAcyclic();
82  		copiesMade.put(this, copy);
83  		List<NfaState> clauseCopies = new ArrayList<>();
84  		for (NfaState clause: clauses) {
85  			clause = clause.copy(null, copiesMade);
86  			clauseCopies.add(clause);
87  		}
88  		copy.clauses.addAll(clauseCopies);
89  		copy.nextState = nextState;
90  		if (nextState == null && dangling != null)
91  			dangling.add(copy);
92  		return copy;
93  	}

95  	@Override
96  	public void setNextState(int input, NfaState state) {
97  		clauses.set(input, state);
98  	}

100 	@Override
101 	public boolean matchesEmptySequence(Set<NfaState> statesVisited) {
102 		if (statesVisited.contains(this)) {
103 			// We've found a cycle. Stop processing, and just return the
104 			// "safest" (least-guarantee) answer. In this case: we can't
105 			// guarantee that this DOESN'T match the empty sequence.
106 			return true;
107 		}
108 		statesVisited.add(this);
109 		boolean anyMatchEmpty = false;
110 		for (NfaState clause: clauses) {
111 			if (clause.matchesEmptySequence(statesVisited)) {
112 				anyMatchEmpty = true;
113 				break;
114 			}
115 		}
116 		if (anyMatchEmpty && (nextState == null || nextState.matchesEmptySequence(statesVisited))) {
117 			return true;
118 		}
119 		return false;
120 	}

145 	@Override
146 	public int hitsLengthMin(Set<NfaState> statesVisited) {
147 		int hitLengthMin = Integer.MAX_VALUE;
148 		if (statesVisited.contains(this)) {
149 			// We've found a cycle. Stop processing, and just return the
150 			// "safest" (least-guarantee) answer. In this case: the smallest
151 			// hit might be 0 long.
152 			return 0;
153 		}
154 		statesVisited.add(this);
155 		for (NfaState clause: clauses) {
156 			int i = clause.hitsLengthMin(statesVisited);
157 			if (i < hitLengthMin)
158 				hitLengthMin = i;
159 		}
160 		if (hitLengthMin < Integer.MAX_VALUE && nextState != null) {
161 			hitLengthMin += nextState.hitsLengthMin(statesVisited);
162 		}
163 		return hitLengthMin;
164 	}

166 	@Override
167 	public int hitsLengthMax(Set<NfaState> statesVisited) {
168 		int hitLengthMax = -1;
169 		if (statesVisited.contains(this)) {
170 			// We've found a cycle. Stop processing, and just return the
171 			// "safest" (least-guarantee) answer. In this case: the largest
172 			// hit might be "infinitely" large.
173 			return Integer.MAX_VALUE;
174 		}
175 		statesVisited.add(this);
176 		for (NfaState clause: clauses) {
177 			int i = clause.hitsLengthMax(statesVisited);
178 			if (i > hitLengthMax)
179 				hitLengthMax = i;
180 		}
181 		if (hitLengthMax >= 0 && nextState != null) {
182 			hitLengthMax += nextState.hitsLengthMax(statesVisited);
183 		}
184 		if (hitLengthMax < 0)
185 			return 0;
186 		return hitLengthMax;
187 	}

189 	@Override
190 	protected String dumpInternal(Map<NfaState, Integer> stateNrs) {
191 		StringBuilder b = new StringBuilder();
192 		for (NfaState clause: clauses) {
193 			if (b.length() > 0)
194 				b.append(",");
195 			b.append(dump(clause, stateNrs));
196 		}
197 		return "OR(" + b.toString() + ", " + dump(nextState, stateNrs) + ")";
198 	}

200 	@Override
201 	void lookupPropertyNumbersInternal(ForwardIndexAccessor fiAccessor, Map<NfaState, Boolean> statesVisited) {
202 		for (NfaState clause: clauses) {
203 			if (clause != null)
204 				clause.lookupPropertyNumbers(fiAccessor, statesVisited);
205 		}
206 		if (nextState != null)
207 			nextState.lookupPropertyNumbers(fiAccessor, statesVisited);
208 	}

210 	@Override
211 	protected void finishInternal(Set<NfaState> visited) {
212 		for (int i = 0; i < clauses.size(); i++) {
213 			NfaState clause = clauses.get(i);
214 			if (clause == null)
215 				clauses.set(i, match());
216 			else
217 				clause.finish(visited);
218 		}
219 		if (nextState == null)
220 			nextState = match();
221 		else
222 			nextState.finish(visited);
223 	}
