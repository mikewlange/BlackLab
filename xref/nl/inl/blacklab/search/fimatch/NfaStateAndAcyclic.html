10  /**
11   * An AND-node where none of the clauses cycle back to an
12   * earlier node. This can be evaluated more efficiently,
13   * by finding matches for each of the clauses, and then
14   * continuing at nextState from each of the matches found.
15   */
16  public class NfaStateAndAcyclic extends NfaState {
17  
18  	List<NfaState> clauses;
19  
20  	NfaState nextState;
21  
22  	public NfaStateAndAcyclic(List<NfaState> andClauses) {
23  		for (NfaState clause: andClauses) {
24  			clause.finish(new HashSet<NfaState>());
25  		}
26  		this.clauses = new ArrayList<>(andClauses);
27  		this.nextState = null;
28  	}
29  
30  	public NfaStateAndAcyclic() {
31  		this.clauses = new ArrayList<>();
32  		this.nextState = null;
33  	}
34  
35  	@Override
36  	public boolean findMatchesInternal(ForwardIndexDocument fiDoc, int pos, int direction, Set<Integer> matchEnds) {
37  		// AND state. Find matches for all alternatives.
38  		Set<Integer> clausesMatchEnds = null;
39  		Set<Integer> matchEndsThisClause = new HashSet<>();
40  		for (NfaState clause: clauses) {
41  			matchEndsThisClause.clear();
42  			if (!clause.findMatchesInternal(fiDoc, pos, direction, matchEndsThisClause))
43  				return false; // this clause had no hits; short circuit AND
44  			if (clausesMatchEnds == null) {
45  				// First matches found
46  				clausesMatchEnds = matchEndsThisClause;
47  				matchEndsThisClause = new HashSet<>();
48  			} else {
49  				// Determine intersection with previous matches
50  				clausesMatchEnds.retainAll(matchEndsThisClause);
51  				if (clausesMatchEnds.size() == 0)
52  					return false; // there are no hits left; short circuit AND
53  			}
54  		}
55  		boolean foundMatch = false;
56  		if (clausesMatchEnds.size() > 0) {
57  			// Continue matching from the matches to our OR clauses
58  			for (Integer clauseMatchEnd: clausesMatchEnds) {
59  				foundMatch |= nextState.findMatchesInternal(fiDoc, clauseMatchEnd, direction, matchEnds);
60  			}
61  		}
62  		return foundMatch;
63  	}
64  
65  	@Override
66  	void fillDangling(NfaState state) {
67  		if (nextState == null)
68  			nextState = state;
69  	}
70  
71  	@Override
72  	NfaStateAndAcyclic copyInternal(Collection<NfaState> dangling, Map<NfaState, NfaState> copiesMade) {
73  		NfaStateAndAcyclic copy = new NfaStateAndAcyclic();
74  		copiesMade.put(this, copy);
75  		List<NfaState> clauseCopies = new ArrayList<>();
76  		for (NfaState clause: clauses) {
77  			clause = clause.copy(null, copiesMade);
78  			clauseCopies.add(clause);
79  		}
80  		copy.clauses.addAll(clauseCopies);
81  		copy.nextState = nextState;
82  		if (nextState == null && dangling != null)
83  			dangling.add(copy);
84  		return copy;
85  	}
86  
87  	@Override
88  	public void setNextState(int input, NfaState state) {
89  		clauses.set(input, state);
90  	}

92  	@Override
93  	public boolean matchesEmptySequence(Set<NfaState> statesVisited) {
94  		if (statesVisited.contains(this)) {
95  			// We've found a cycle. Stop processing, and just return the
96  			// "safest" (least-guarantee) answer. In this case: we can't
97  			// guarantee that this DOESN'T match the empty sequence.
98  			return true;
99  		}
100 		statesVisited.add(this);
101 		for (NfaState clause: clauses) {
102 			if (!clause.matchesEmptySequence(statesVisited))
103 				return false;
104 		}
105 		return nextState == null || nextState.matchesEmptySequence(statesVisited);
106 	}

108 	@Override
109 	public boolean hitsAllSameLength(Set<NfaState> statesVisited) {
110 		if (statesVisited.contains(this)) {
111 			// We've found a cycle. Stop processing, and just return the
112 			// "safest" (least-guarantee) answer. In this case: we can't
113 			// guarantee that hits are all the same length.
114 			return false;
115 		}
116 		statesVisited.add(this);
117 		boolean anyClauseHitsAllSameLength = false;
118 		for (NfaState clause: clauses) {
119 			if (clause.hitsAllSameLength(statesVisited)) {
120 				anyClauseHitsAllSameLength = true;
121 				break;
122 			}
123 		}
124 		return anyClauseHitsAllSameLength && (nextState == null || nextState.hitsAllSameLength(statesVisited));
125 	}
126 
127 	@Override
128 	public int hitsLengthMin(Set<NfaState> statesVisited) {
129 		if (statesVisited.contains(this)) {
130 			// We've found a cycle. Stop processing, and just return the
131 			// "safest" (least-guarantee) answer. In this case: the smallest
132 			// hit might be 0 long.
133 			return 0;
134 		}
135 		statesVisited.add(this);
136 		int hitLengthMin = 0;
137 		for (NfaState nextState: clauses) {
138 			int i = nextState.hitsLengthMin(statesVisited);
139 			if (i > hitLengthMin)
140 				hitLengthMin = i;
141 		}
142 		if (hitLengthMin < Integer.MAX_VALUE && nextState != null) {
143 			hitLengthMin += nextState.hitsLengthMin(statesVisited);
144 		}
145 		return hitLengthMin;
146 	}
147 
148 	@Override
149 	public int hitsLengthMax(Set<NfaState> statesVisited) {
150 		if (statesVisited.contains(this)) {
151 			// We've found a cycle. Stop processing, and just return the
152 			// "safest" (least-guarantee) answer. In this case: the largest
153 			// hit might be "infinitely" large.
154 			return Integer.MAX_VALUE;
155 		}
156 		statesVisited.add(this);
157 		int hitLengthMax = Integer.MAX_VALUE;
158 		for (NfaState nextState: clauses) {
159 			int i = nextState.hitsLengthMax(statesVisited);
160 			if (i < hitLengthMax)
161 				hitLengthMax = i;
162 		}
163 		if (nextState != null) {
164 			hitLengthMax += nextState.hitsLengthMax(statesVisited);
165 		}
166 		return hitLengthMax;
167 	}

169 	@Override
170 	protected String dumpInternal(Map<NfaState, Integer> stateNrs) {
171 		StringBuilder b = new StringBuilder();
172 		for (NfaState s: clauses) {
173 			if (b.length() > 0)
174 				b.append(",");
175 			b.append(dump(s, stateNrs));
176 		}
177 		return "AND(" + b.toString() + ", " + dump(nextState, stateNrs) + ")";
178 	}

180 	@Override
181 	void lookupPropertyNumbersInternal(ForwardIndexAccessor fiAccessor, Map<NfaState, Boolean> statesVisited) {
182 		for (NfaState s: clauses) {
183 			if (s != null)
184 				s.lookupPropertyNumbers(fiAccessor, statesVisited);
185 		}
186 		if (nextState != null)
187 			nextState.lookupPropertyNumbers(fiAccessor, statesVisited);
188 	}
189 
190 	@Override
191 	protected void finishInternal(Set<NfaState> visited) {
192 		for (int i = 0; i < clauses.size(); i++) {
193 			NfaState s = clauses.get(i);
194 			if (s == null)
195 				clauses.set(i, match());
196 			else
197 				s.finish(visited);
198 		}
199 		if (nextState == null)
200 			nextState = match();
201 		else
202 			nextState.finish(visited);
203 	}
