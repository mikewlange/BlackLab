90  		/** Open tag: start indexing this document */
91  		@Override
92  		public void startElement(String uri, String localName, String qName,
93  				Attributes attributes) {
94  			startCaptureContent(contentsField.getName());
95  
96  			currentLuceneDoc = new Document();
97  			currentDocumentName = fileName;
98  			if (currentDocumentName == null)
99  				currentDocumentName = "?";
100 			// Store attribute values from the tag as metadata fields
101 			for (int i = 0; i < attributes.getLength(); i++) {
102 				addMetadataField(attributes.getLocalName(i),
103 						attributes.getValue(i));
104 			}
105 			currentLuceneDoc.add(new Field("fromInputFile", currentDocumentName, indexer.metadataFieldTypeUntokenized));
106 			addMetadataFieldsFromParameters();
107 			indexer.getListener().documentStarted(currentDocumentName);
108 		}
109 
110 		/** Open tag: end indexing the document */
111 		@Override
112 		public void endElement(String uri, String localName, String qName) {
113 			// Make sure all the properties have an equal number of values.
114 			// See what property has the highest position
115 			// (in practice, only starttags and endtags should be able to have
116 			// a position one higher than the rest)
117 			int lastValuePos = 0;
118 			for (ComplexFieldProperty prop : contentsField.getProperties()) {
119 				if (prop.lastValuePosition() > lastValuePos)
120 					lastValuePos = prop.lastValuePosition();
121 			}
122 
123 			// Make sure we always have one more token than the number of
124 			// words, so there's room for any tags after the last word, and we
125 			// know we should always skip the last token when matching.
126 			if (propMain.lastValuePosition() == lastValuePos)
127 				lastValuePos++;
128 
129 			// Add empty values to all lagging properties
130 			for (ComplexFieldProperty prop : contentsField.getProperties()) {
131 				while (prop.lastValuePosition() < lastValuePos) {
132 					prop.addValue("");
133 					if (prop.hasPayload())
134 						prop.addPayload(null);
135 					if (prop == propMain) {
136 						contentsField.addStartChar(getCharacterPosition());
137 						contentsField.addEndChar(getCharacterPosition());
138 					}
139 				}
140 			}
141 
142 			// Finish storing the document in the document store (parts of it
143 			// may already have been written because we write in chunks to save memory),
144 			// retrieve the content id, and store that in Lucene.
145 			// (Note that we do this after adding the dummy token, so the character
146 			//  positions for the dummy token still make (some) sense)
147 			int contentId = storeCapturedContent();
148 			currentLuceneDoc.add(new IntField(ComplexFieldUtil
149 					.contentIdField(contentsField.getName()), contentId,
150 					Store.YES));
151 
152 			// Store the different properties of the complex contents field that
153 			// were gathered in
154 			// lists while parsing.
155 			contentsField.addToLuceneDoc(currentLuceneDoc);
156 
157 
158 			// Add all properties to forward index
159 			for (ComplexFieldProperty prop : contentsField.getProperties()) {
160 				if (!prop.hasForwardIndex())
161 					continue;
162 
163 				// Add property (case-sensitive tokens) to forward index and add
164 				// id to Lucene doc
165 				String propName = prop.getName();
166 				String fieldName = ComplexFieldUtil.propertyField(
167 						contentsField.getName(), propName);
168 				int fiid = indexer.addToForwardIndex(fieldName, prop);
169 				currentLuceneDoc.add(new IntField(ComplexFieldUtil
170 						.forwardIndexIdField(fieldName), fiid, Store.YES));
171 			}
172 
173 			// If there's an external metadata fetcher, call it now so it can
174 			// add the metadata for this document and (optionally) store the
175 			// metadata
176 			// document in the content store (and the corresponding id in the
177 			// Lucene doc)
178 			MetadataFetcher m = getMetadataFetcher();
179 			if (m != null) {
180 				m.addMetadata();
181 			}
182 
183 			// See what metadatafields are missing or empty and add unknown value
184 			// if desired.
185 			IndexStructure struct = indexer.getSearcher().getIndexStructure();
186 			for (String fieldName: struct.getMetadataFields()) {
187 				MetadataFieldDesc fd = struct.getMetadataFieldDesc(fieldName);
188 				boolean missing = false, empty = false;
189 				String currentValue = currentLuceneDoc.get(fieldName);
190 				if (currentValue == null)
191 					missing = true;
192 				else if (currentValue.length() == 0)
193 					empty = true;
194 				UnknownCondition cond = fd.getUnknownCondition();
195 				boolean useUnknownValue = false;
196 				switch(cond) {
197 				case EMPTY:
198 					useUnknownValue = empty;
199 					break;
200 				case MISSING:
201 					useUnknownValue = missing;
202 					break;
203 				case MISSING_OR_EMPTY:
204 					useUnknownValue = missing | empty;
205 					break;
206 				case NEVER:
207 					useUnknownValue = false;
208 					break;
209 				}
210 				if (useUnknownValue)
211 					addMetadataField(fieldName, fd.getUnknownValue());
212 			}
213 
214 			try {
215 				// Add Lucene doc to indexer
216 				indexer.add(currentLuceneDoc);
217 			} catch (Exception e) {
218 				throw ExUtil.wrapRuntimeException(e);
219 			}
220 
221 			// Report progress
222 			reportCharsProcessed();
223 			reportTokensProcessed(wordsDone);
224 			wordsDone = 0;
225 			indexer.getListener().documentDone(currentDocumentName);
226 
227 			// Reset contents field for next document
228 			contentsField.clear();
229 			currentLuceneDoc = null;
230 
231 			// Stop if required
232 			if (!indexer.continueIndexing())
233 				throw new MaxDocsReachedException();
234 		}
235 	}

237 	/**
238 	 * If any metadata fields were supplied in the indexer parameters,
239 	 * add them now.
240 	 * 
241 	 * NOTE: we always add these untokenized (because they're usually just
242 	 * indications of which data set a set of files belongs to), but that
243 	 * means they don't get lowercased or de-accented. Because metadata queries
244 	 * are always desensitized, you can't use uppercase or accented letters in
245 	 * these values or they will never be found. This should be addressed.
246 	 */
247 	void addMetadataFieldsFromParameters() {
248 		for (Entry<String, String> e: parameters.entrySet()) {
249 			if (e.getKey().startsWith("meta-")) {
250 				String fieldName = e.getKey().substring(5);
251 				String fieldValue = e.getValue();
252 				currentLuceneDoc.add(new Field(fieldName, fieldValue, indexer.metadataFieldTypeUntokenized));
253 			}
254 		}
255 	}

336 	/** Handle tags. */
337 	public class InlineTagHandler extends ElementHandler {
338 
339 		/** Open tag: store the start tag location and the attribute values */
340 		@Override
341 		public void startElement(String uri, String localName, String qName,
342 				Attributes attributes) {
343 			int lastStartTagPos = propStartTag.lastValuePosition();
344 			int currentPos = propMain.lastValuePosition() + 1;
345 			int posIncrement = currentPos - lastStartTagPos;
346 			propStartTag.addValue(localName, posIncrement);
347 			propStartTag.addPayload(null);
348 			int startTagIndex = propStartTag.getLastValueIndex();
349 			openTags.add(new OpenTagInfo(currentPos, startTagIndex));
350 			for (int i = 0; i < attributes.getLength(); i++) {
351 				// Index element attribute values
352 				String name = attributes.getLocalName(i);
353 				String value = attributes.getValue(i);
354 				propStartTag.addValue(
355 						"@" + name.toLowerCase() + "__" + value.toLowerCase(),
356 						0);
357 				propStartTag.addPayload(null);
358 			}
359 		}
360 
361 		/** Close tag: store the end tag location */
362 		@Override
363 		public void endElement(String uri, String localName, String qName) {
364 			int currentPos = propMain.lastValuePosition() + 1;
365 
366 			// Add payload to start tag property indicating end position
367 			OpenTagInfo openTag = openTags.remove(openTags.size() - 1);
368 			byte[] payload = ByteBuffer.allocate(4).putInt(currentPos).array();
369 			propStartTag.setPayloadAtIndex(openTag.index, new BytesRef(payload));
370 		}
371 	}

485 	SensitivitySetting getSensitivitySetting(String propName) {
486 		// See if it's specified in a parameter
487 		String strSensitivity = getParameter(propName + "_sensitivity");
488 		if (strSensitivity != null) {
489 			if (strSensitivity.equals("i"))
490 				return SensitivitySetting.ONLY_INSENSITIVE;
491 			if (strSensitivity.equals("s"))
492 				return SensitivitySetting.ONLY_SENSITIVE;
493 			if (strSensitivity.equals("si") || strSensitivity.equals("is"))
494 				return SensitivitySetting.SENSITIVE_AND_INSENSITIVE;
495 			if (strSensitivity.equals("all"))
496 				return SensitivitySetting.CASE_AND_DIACRITICS_SEPARATE;
497 		}
498 
499 		// Not in parameter (or unrecognized value), use default based on
500 		// propName
501 		if (propName.equals(ComplexFieldUtil.getDefaultMainPropName())
502 				|| propName.equals(ComplexFieldUtil.LEMMA_PROP_NAME)) {
503 			// Word: default to sensitive/insensitive
504 			return SensitivitySetting.SENSITIVE_AND_INSENSITIVE;
505 		}
506 		if (propName.equals(ComplexFieldUtil.PUNCTUATION_PROP_NAME)) {
507 			// Punctuation: default to only insensitive
508 			return SensitivitySetting.ONLY_INSENSITIVE;
509 		}
510 		if (propName.equals(ComplexFieldUtil.START_TAG_PROP_NAME)
511 				|| propName.equals(ComplexFieldUtil.END_TAG_PROP_NAME)) {
512 			// XML tag properties: default to only sensitive
513 			return SensitivitySetting.ONLY_SENSITIVE;
514 		}
515 
516 		// Unrecognized; default to only insensitive
517 		return SensitivitySetting.ONLY_INSENSITIVE;
518 	}

553 	/**
554 	 * StringBuffer re-used for building start/end tags and processing
555 	 * instructions.
556 	 */
557 	StringBuilder elementBuilder = new StringBuilder();
558 
559 	public void startElement(String uri, String localName, String qName,
560 			Attributes attributes) {
561 		// Call any hooks associated with this element
562 		hookableHandler.startElement(uri, localName, qName, attributes);
563 
564 		elementBuilder.setLength(0); // clear
565 		elementBuilder.append("<").append(qName);
566 		for (int i = 0; i < attributes.getLength(); i++) {
567 			String value = escapeXmlChars(attributes.getValue(i));
568 			elementBuilder.append(" ").append(attributes.getQName(i))
569 					.append("=\"").append(value).append("\"");
570 		}
571 		// Append any namespace mapping not yet outputted
572 		if (!outputPrefixMapping.isEmpty()) {
573 			for (Map.Entry<String, String> e : outputPrefixMapping.entrySet()) {
574 				if (e.getKey().length() == 0)
575 					elementBuilder.append(" xmlns=\"").append(e.getValue())
576 							.append("\"");
577 				else
578 					elementBuilder.append(" xmlns:").append(e.getKey())
579 							.append("=\"").append(e.getValue()).append("\"");
580 			}
581 			outputPrefixMapping.clear(); // outputted all prefix mappings for
582 											// now
583 		}
584 		elementBuilder.append(">");
585 		processContent(elementBuilder.toString());
586 	}

590 	 */
591 	StringBuilder escapeBuilder = new StringBuilder();
592 
593 	/**
594 	 * Escape the special XML chars (<, >, &, ") with their named entity
595 	 * equivalents.
596 	 *
597 	 * NOTE: copy of StringUtil.escapeXmlChars that re-uses its StringBuilder
598 	 * for increased memory efficiency.
599 	 *
600 	 * @param source
601 	 *            the source string
602 	 * @return the escaped string
603 	 */
604 	public String escapeXmlChars(String source) {
605 		escapeBuilder.setLength(0); // clear
606 		int start = 0;
607 		char[] srcArr = new char[source.length()];
608 		source.getChars(0, source.length(), srcArr, 0);
609 		int end = source.length();
610 		for (int i = 0; i < end; i++) {
611 			char c = srcArr[i]; // source.charAt(i);
612 			if (c == '<' || c == '>' || c == '&' || c == '"') {
613 				escapeBuilder.append(srcArr, start, i - start); // source.substring(start,
614 																// i));
615 				switch (c) {
616 				case '<':
617 					escapeBuilder.append("&lt;");
618 					break;
619 				case '>':
620 					escapeBuilder.append("&gt;");
621 					break;
622 				case '&':
623 					escapeBuilder.append("&amp;");
624 					break;
625 				case '"':
626 					escapeBuilder.append("&quot;");
627 					break;
628 				}
629 				start = i + 1;
630 			}
631 		}
632 		escapeBuilder.append(srcArr, start, end - start); // source.substring(start));
633 		return escapeBuilder.toString();
634 	}

638 	 * equivalents.
639 	 *
640 	 * NOTE: copy of StringUtil.escapeXmlChars that re-uses its StringBuilder
641 	 * for increased memory efficiency.
642 	 *
643 	 * @param source
644 	 *            the source string
645 	 * @param start
646 	 *            start index of the string to escape
647 	 * @param length
648 	 *            length of the string to escape
649 	 * @return the escaped string
650 	 */
651 	public String escapeXmlChars(char[] source, int start, int length) {
652 		escapeBuilder.setLength(0); // clear
653 		int end = start + length;
654 		for (int i = start; i < end; i++) {
655 			char c = source[i];
656 			if (c == '<' || c == '>' || c == '&' || c == '"') {
657 				escapeBuilder.append(source, start, i - start);
658 				switch (c) {
659 				case '<':
660 					escapeBuilder.append("&lt;");
661 					break;
662 				case '>':
663 					escapeBuilder.append("&gt;");
664 					break;
665 				case '&':
666 					escapeBuilder.append("&amp;");
667 					break;
668 				case '"':
669 					escapeBuilder.append("&quot;");
670 					break;
671 				}
672 				start = i + 1;
673 			}
674 		}
675 		escapeBuilder.append(source, start, end - start);
676 		return escapeBuilder.toString();
677 	}

782 	 *
783 	 * @return the metadata fetcher if any, or null if there is none.

785 	MetadataFetcher getMetadataFetcher() {
786 		if (metadataFetcher == null) {
787 			String metadataFetcherClassName = getParameter("metadataFetcherClass");
788 			if (metadataFetcherClassName != null) {

790 					Class<? extends MetadataFetcher> metadataFetcherClass = Class.forName(metadataFetcherClassName).asSubclass(MetadataFetcher.class);
791 					Constructor<? extends MetadataFetcher> ctor = metadataFetcherClass.getConstructor(DocIndexer.class);
792 					metadataFetcher = ctor.newInstance(this);
793 				} catch (Exception e) {
794 					throw new RuntimeException(e);

833 	public void addMetadataField(String name, String value) {
834 
835 		IndexStructure struct = indexer.getSearcher().getIndexStructure();
836 		struct.registerMetadataField(name);
837 
838 		MetadataFieldDesc desc = struct.getMetadataFieldDesc(name);
839 		FieldType type = desc.getType();
840 		desc.addValue(value);
841 
842 		FieldType shouldBeType = getMetadataFieldTypeFromIndexerProperties(name);
843 		if (type == FieldType.TEXT
844 				&& shouldBeType != FieldType.TEXT) {
845 			// indexer.properties overriding default type
846 			type = shouldBeType;
847 		}
848 		if (type != FieldType.NUMERIC) {
849 			currentLuceneDoc.add(new Field(name, value,
850 					luceneTypeFromIndexStructType(type)));
851 		}
852 		if (type == FieldType.NUMERIC || numericFields.contains(name)) {
853 			String numFieldName = name;
854 			if (type != FieldType.NUMERIC) {
855 				numFieldName += "Numeric";
856 			}
857 			// Index these fields as numeric too, for faster range queries
858 			// (we do both because fields sometimes aren't exclusively numeric)
859 			int n = 0;
860 			try {
861 				n = Integer.parseInt(value);
862 			} catch (NumberFormatException e) {
863 				// This just happens sometimes, e.g. given multiple years, or
864 				// descriptive text like "around 1900". OK to ignore.
865 			}
866 			IntField nf = new IntField(numFieldName, n, Store.YES);
867 			currentLuceneDoc.add(nf);
868 		}
869 	}

904 	@Override
905 	public void index() throws IOException, InputFormatException {
906 		SAXParserFactory factory = SAXParserFactory.newInstance();
907 		factory.setNamespaceAware(true);
908 		SAXParser parser;
909 		try {
910 			parser = factory.newSAXParser();
911 		} catch (Exception e1) {
912 			// Unrecoverable error, throw runtime exception
913 			throw new RuntimeException(e1);
914 		}
915 		try {
916 			InputSource is = new InputSource(reader);
917 			XMLReader xmlReader = parser.getXMLReader();
918 			xmlReader.setEntityResolver(new DummyEntityResolver());
919 			xmlReader.setContentHandler(saxParseHandler);
920 			xmlReader.parse(is);
921 		} catch (IOException e) {
922 			throw e;
923 		} catch (SAXException e) {
924 			throw new InputFormatException(e);
925 		} catch (DocIndexer.MaxDocsReachedException e) {
926 			// OK; just stop indexing prematurely
927 		}
928 
929 		if (nDocumentsSkipped > 0)
930 			System.err.println("Skipped " + nDocumentsSkipped
931 					+ " large documents");
932 	}
