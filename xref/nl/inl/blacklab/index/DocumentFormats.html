6   import java.util.Map;

9   import nl.inl.blacklab.indexers.DocIndexerAlto;
10  import nl.inl.blacklab.indexers.DocIndexerFolia;
11  import nl.inl.blacklab.indexers.DocIndexerPageXml;
12  import nl.inl.blacklab.indexers.DocIndexerTei;
13  import nl.inl.blacklab.indexers.DocIndexerTeiPosInFunctionAttr;
14  import nl.inl.blacklab.indexers.DocIndexerTeiText;

17  /**
18   * Document format registry, for resolving a DocIndexer class given a
19   * format identifier (common abbreviation or (qualified) class name).
20   */
21  public class DocumentFormats {
22  
23  	/** Document formats */
24  	static Map<String, Class<? extends DocIndexer>> formats = new TreeMap<>();
25  
26  	static {
27  		// Some abbreviations for commonly used builtin DocIndexers.
28  		// You can also specify the classname for builtin DocIndexers,
29  		// or a fully-qualified name for your custom DocIndexer (must
30  		// be on the classpath)
31  		register("alto", DocIndexerAlto.class);
32  		register("folia", DocIndexerFolia.class);
33  		register("pagexml", DocIndexerPageXml.class);
34  		register("sketchxml", DocIndexerXmlSketch.class);
35  
36  		// TEI has a number of variants
37  		// By default, the contents of the "body" element are indexed, but alternatively you can index the contents of "text".
38  		// By default, the "type" attribute is assumed to contain PoS, but alternatively you can use the "function" attribute.
39  		register("tei", DocIndexerTei.class);
40  		register("tei-element-body", DocIndexerTei.class);
41  		register("tei-element-text", DocIndexerTeiText.class);
42  		register("tei-pos-type", DocIndexerTei.class);
43  		register("tei-pos-function", DocIndexerTeiPosInFunctionAttr.class);
44  	}
45  
46  	/**
47  	 * Register an abbreviated document format identifier.
48  	 *
49  	 * @param formatAbbreviation the format abbreviation, e.g. "tei"
50  	 *   (NOTE: format abbreviations are case-insensitive, and are lowercased internally)
51  	 * @param docIndexerClass the DocIndexer class for this format
52  	 */
53  	public static void register(String formatAbbreviation, Class<? extends DocIndexer> docIndexerClass) {
54  		formats.put(formatAbbreviation.toLowerCase(), docIndexerClass);
55  	}
56  
57  	/**
58  	 * Get the DocIndexer class associated with the format identifier.
59  	 *
60  	 * @param formatIdentifier format identifier, e.g. "tei" or "com.example.MyIndexer"
61  	 * @return the DocIndexer class, or null if not found
62  	 */
63  	@SuppressWarnings("unchecked")
64  	public static Class<? extends DocIndexer> getIndexerClass(String formatIdentifier) {
65  		// Check if it's a known abbreviation.
66  		Class<? extends DocIndexer> docIndexerClass = formats.get(formatIdentifier.toLowerCase());
67  		if (docIndexerClass == null) {
68  			// No; is it a fully qualified class name?
69  			try {
70  				docIndexerClass = (Class<? extends DocIndexer>) Class.forName(formatIdentifier);
71  			} catch (Exception e1) {
72  				try {
73  					// No. Is it a class in the BlackLab indexers package?
74  					docIndexerClass = (Class<? extends DocIndexer>) Class
75  							.forName("nl.inl.blacklab.indexers." + formatIdentifier);
76  				} catch (Exception e) {
77  					// Couldn't be resolved. That's okay, we'll just return null and let
78  					// the application deal with it.
79  				}
80  			}
81  		}
82  		return docIndexerClass;
83  	}
84  
85  	/**
86  	 * Check if a particular string denotes a valid document format.
87  	 *
88  	 * @param formatIdentifier format identifier, e.g. "tei" or "com.example.MyIndexer"
89  	 * @return true iff it corresponds to a format
90  	 */
91  	public static boolean exists(String formatIdentifier) {
92  		return getIndexerClass(formatIdentifier) != null;
93  	}
94  
95  	/**
96  	 * Returns a sorted list of registered document format abbreviations.
97  	 *
98  	 * @return the list of registered abbreviations
99  	 */
100 	public static List<String> list() {
101 		List<String> l = new ArrayList<>(formats.keySet());
102 		Collections.sort(l);
103 		return Collections.unmodifiableList(l);
104 	}
105 
106 }
