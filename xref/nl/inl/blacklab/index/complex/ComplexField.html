26  import org.apache.lucene.document.IntField;
27  import org.eclipse.collections.impl.list.mutable.primitive.IntArrayList;

32  /**
33   * A complex field is like a Lucene field, but in addition to its "normal" value, it can have
34   * multiple properties per word (not just a single token). The properties might be "headword", "pos"
35   * (part of speech), "namedentity" (whether or not the word is (part of) a named entity like a
36   * location or place), etc.
37   *
38   * Complex fields are implemented by indexing a field in Lucene for each property. For example, if
39   * complex field "contents" has properties "headword" and "pos", there would be 3 Lucene fields for
40   * the complex field: "contents", "contents__headword" and "contents__pos".
41   *
42   * The main field ("contents" in the above example) may include offset information if you want (e.g.
43   * for highlighting). All Lucene fields will include position information (for use with
44   * SpanQueries).
45   *
46   * N.B. It is crucial that everything stays in synch, so you should call all the appropriate add*()
47   * methods for each property and each token, or use the correct position increments to keep everything
48   * synched up. The same goes for addStartChar() and addEndChar() (although, if you don't want any
49   * offsets, you need not call these).
50   */
51  public class ComplexField {
52  
53  	private Map<String, ComplexFieldProperty> properties = new HashMap<>();
54  
55  	private IntArrayList start = new IntArrayList();
56  
57  	private IntArrayList end = new IntArrayList();
58  
59  	private String fieldName;
60  
61  	private ComplexFieldProperty mainProperty;
62  
63  	private Set<String> noForwardIndexProps = Collections.emptySet();
64  
65  	public void setNoForwardIndexProps(Set<String> noForwardIndexProps) {
66  		this.noForwardIndexProps = noForwardIndexProps;
67  	}
68  
69  	/**
70  	 * Construct a ComplexField object with a main property
71  	 * @param name field name
72  	 * @param mainPropertyName main property name
73  	 * @param sensitivity ways to index main property, with respect to case- and
74  	 *   diacritics-sensitivity.
75  	 * @param mainPropHasPayloads does the main property have payloads?
76  	 */
77  	public ComplexField(String name, String mainPropertyName, SensitivitySetting sensitivity, boolean mainPropHasPayloads) {
78  		boolean includeOffsets = true;
79  		fieldName = name;
80  		if (mainPropertyName == null)
81  			mainPropertyName = ComplexFieldUtil.getDefaultMainPropName();
82  		mainProperty = new ComplexFieldProperty(mainPropertyName, sensitivity, includeOffsets, mainPropHasPayloads);
83  		properties.put(mainPropertyName, mainProperty);
84  	}
85  
86  	public int numberOfTokens() {
87  		return start.size();
88  	}
89  
90  	public ComplexFieldProperty addProperty(String name, SensitivitySetting sensitivity, boolean includePayloads) {
91  		ComplexFieldProperty p = new ComplexFieldProperty(name, sensitivity, false, includePayloads);
92  		if (noForwardIndexProps.contains(name)) {
93  			p.setForwardIndex(false);
94  		}
95  		properties.put(name, p);
96  		return p;
97  	}
98  
99  	public ComplexFieldProperty addProperty(String name, SensitivitySetting sensitivity) {
100 		return addProperty(name, sensitivity, false);
101 	}
102 
103 	public void addStartChar(int startChar) {
104 		start.add(startChar);
105 	}
106 
107 	public void addEndChar(int endChar) {
108 		end.add(endChar);
109 	}
110 
111 	public void addToLuceneDoc(Document doc) {
112 		for (ComplexFieldProperty p : properties.values()) {
113 			p.addToLuceneDoc(doc, fieldName, start, end);
114 		}
115 
116 		// Add number of tokens in complex field as a stored field,
117 		// because we need to be able to find this property quickly
118 		// for SpanQueryNot.
119 		// (Also note that this is the actual number of words + 1,
120 		//  because we always store a dummy "closing token" at the end
121 		//  that doesn't contain a word but may contain trailing punctuation)
122 		doc.add(new IntField(ComplexFieldUtil.lengthTokensField(fieldName), numberOfTokens(), Field.Store.YES));
123 	}
124 
125 	public void clear() {
126 		start.clear();
127 		end.clear();
128 		for (ComplexFieldProperty p : properties.values()) {
129 			p.clear();
130 		}
131 	}
132 
133 	public ComplexFieldProperty getProperty(String name) {
134 		ComplexFieldProperty p = properties.get(name);
135 		if (p == null)
136 			throw new IllegalArgumentException("Undefined property '" + name + "'");
137 		return p;
138 	}
139 
140 	public ComplexFieldProperty getMainProperty() {
141 		return mainProperty;
142 	}
143 
144 	public String getName() {
145 		return fieldName;
146 	}
147 
148 	public Collection<ComplexFieldProperty> getProperties() {
149 		return properties.values();
150 	}
151 
152 
153 }
