33  /**
34   * Analyzer that doesn't tokenize but returns a single token.
35   *
36   * Has the option of analyzing case-/accent-sensitive or -insensitive, depending on the field name.
37   */
38  public final class BLNonTokenizingAnalyzer extends Analyzer {
39  
40  	@Override
41  	protected TokenStreamComponents createComponents(String fieldName, Reader reader) {
42  		try {
43  			Tokenizer source = new BLNonTokenizer(reader);
44  			source.reset();
45  			TokenStream filter = source;
46  			boolean caseSensitive = ComplexFieldUtil.isCaseSensitive(fieldName);
47  			if (!caseSensitive)
48  			{
49  				filter = new LowerCaseFilter(Version.LUCENE_42, filter);// lowercase all
50  				filter.reset();
51  			}
52  			boolean diacSensitive = ComplexFieldUtil.isDiacriticsSensitive(fieldName);
53  			if (!diacSensitive)
54  			{
55  				filter = new RemoveAllAccentsFilter(filter); // remove accents
56  				filter.reset();
57  			}
58  			return new TokenStreamComponents(source, filter);
59  		} catch (IOException e) {
60  			throw new RuntimeException(e);
61  		}
62  	}
