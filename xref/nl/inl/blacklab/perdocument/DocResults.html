151 	/**
152 	 * Construct DocResults from a Scorer (Lucene document results).
153 	 *
154 	 * @param searcher the searcher that generated the results
155 	 * @param scorer the scorer to read document results from
156 	 */
157 	DocResults(Searcher searcher, Scorer scorer) {
158 		this.searcher = searcher;
159 		if (scorer == null)
160 			return; // no matches, empty result set
161 		try {
162 			IndexReader indexReader = searcher.getIndexReader();
163 			while (true) {
164 				int docId;
165 				try {
166 					docId = scorer.nextDoc();
167 				} catch (IOException e) {
168 					throw new RuntimeException(e);
169 				}
170 				if (docId == DocIdSetIterator.NO_MORE_DOCS)
171 					break;
172 
173 				Document d = indexReader.document(docId);
174 				DocResult dr = new DocResult(searcher, null, docId, d, scorer.score());
175 				results.add(dr);
176 			}
177 		} catch (Exception e) {
178 			throw new RuntimeException(e);
179 		}
180 	}

274 	/**
275 	 * Determines if there are at least a certain number of results
276 	 *
277 	 * This may be used if we don't want to process all results (which
278 	 * may be a lot) but we do need to know something about the size
279 	 * of the result set (such as for paging).
280 	 *
281 	 * @param lowerBound the number we're testing against
282 	 *
283 	 * @return true if the size of this set is at least lowerBound, false otherwise.
284 	 */
285 	public boolean sizeAtLeast(int lowerBound) {
286 		try {
287 			// Try to fetch at least this many hits
288 			ensureResultsRead(lowerBound);
289 		} catch (InterruptedException e) {
290 			// Thread was interrupted; abort operation
291 			// and let client decide what to do
292 		}
293 
294 		return results.size() >= lowerBound;
295 	}

299 	 *

305 	 * @return the number of documents.

347 	 * @param toIndex first hit not to include in the resulting list
348 	 * @return the sublist

351 		try {

353 		} catch (InterruptedException e) {

361 	/**
362 	 * If we still have only partially read our Hits object,
363 	 * read the rest of it and add all the hits.
364 	 * @throws InterruptedException
365 	 */
366 	private void ensureAllResultsRead() throws InterruptedException {
367 		ensureResultsRead(-1);
368 	}
369 
370 	/**
371 	 * If we still have only partially read our Hits object,
372 	 * read some more of it and add the hits.
373 	 *
374 	 * @param index the number of results we want to ensure have been read, or negative for all results
375 	 * @throws InterruptedException
376 	 */
377 	synchronized void ensureResultsRead(int index) throws InterruptedException {
378 		if (sourceHitsFullyRead())
379 			return;
380 
381 		try {
382 			synchronized(sourceHitsIterator) {
383 				// Fill list of document results
384 				int doc = partialDocId;
385 				List<Hit> docHits = partialDocHits;
386 				partialDocId = -1;
387 				partialDocHits = null;
388 
389 				IndexReader indexReader = searcher == null ? null : searcher.getIndexReader();
390 				//Thread currentThread = Thread.currentThread();
391 				while ( (index < 0 || results.size() <= index) && sourceHitsIterator.hasNext()) {
392 
393 					Hit hit = sourceHitsIterator.next();
394 					if (hit.doc != doc) {
395 						if (docHits != null) {
396 							Hits hits = new Hits(searcher, docHits);
397 							hits.copySettingsFrom(sourceHits); // concordance type, etc.
398 							addDocResultToList(doc, hits, indexReader);
399 						}
400 						doc = hit.doc;
401 						docHits = new ArrayList<>();
402 					}
403 					docHits.add(hit);
404 				}
405 				// add the final dr instance to the results collection
406 				if (docHits != null) {
407 					if (sourceHitsIterator.hasNext()) {
408 						partialDocId = doc;
409 						partialDocHits = docHits; // not done, continue from here later
410 					} else {
411 						Hits hits = new Hits(searcher, docHits);
412 						hits.copySettingsFrom(sourceHits); // concordance type, etc.
413 						addDocResultToList(doc, hits, indexReader);
414 					}
415 				}
416 			}
417 		} catch (IOException e) {
418 			throw new RuntimeException(e);
419 		}
420 	}
421 
422 	private void addDocResultToList(int doc, Hits docHits, IndexReader indexReader) throws IOException {
423 		DocResult docResult = new DocResult(searcher, sourceHits.getConcordanceFieldName(), doc, indexReader == null ? null : indexReader.document(doc), docHits);
424 		// Make sure we remember what kind of context we have, if any
425 		docResult.setContextField(sourceHits.getContextFieldPropName());
426 		results.add(docResult);
427 	}

459 	/**
460 	 * Return an iterator over these hits.
461 	 *
462 	 * @return the iterator
463 	 */
464 	@Override
465 	public Iterator<DocResult> iterator() {
466 		// Construct a custom iterator that iterates over the hits in the hits
467 		// list, but can also take into account the Spans object that may not have
468 		// been fully read. This ensures we don't instantiate Hit objects for all hits
469 		// if we just want to display the first few.
470 		return new Iterator<DocResult>() {
471 
472 			int index = -1;
473 
474 			@Override
475 			public boolean hasNext() {
476 				// Do we still have hits in the hits list?
477 				try {
478 					ensureResultsRead(index + 1);
479 				} catch (InterruptedException e) {
480 					// Thread was interrupted. Act like we're done.
481 					// Let caller detect and deal with interruption.
482 					return false;
483 				}
484 				return index + 1 < results.size();
485 			}
486 
487 			@Override
488 			public DocResult next() {
489 				// Check if there is a next, taking unread hits from Spans into account
490 				if (hasNext()) {
491 					index++;
492 					return results.get(index);
493 				}
494 				throw new NoSuchElementException();
495 			}
496 
497 			@Override
498 			public void remove() {
499 				throw new UnsupportedOperationException();
500 			}
501 
502 		};
503 	}

505 	public DocResult get(int i) {
506 		try {
507 			ensureResultsRead(i);
508 		} catch (InterruptedException e) {
509 			// Thread was interrupted. Required hit hasn't been gathered;
510 			// we will just return null.
511 		}
512 		if (i >= results.size())
513 			return null;
514 		return results.get(i);
515 	}

554 	/**
555 	 * Sum a property for all the documents.
556 	 *
557 	 * Can be used to calculate the total number of tokens in a subcorpus, for example.
558 	 * Note that this does retrieve all results, so it may be slow for large sets.
559 	 * In particular, you should try to call this method only for DocResults created with
560 	 * Searcher.queryDocuments() (and not ones created with Hits.perDocResults()) to avoid
561 	 * the overhead of fetching hits.
562 	 *
563 	 * @param numProp a numeric property to sum
564 	 * @return the sum
565 	 */
566 	public int intSum(DocProperty numProp) {
567 		try {
568 			ensureAllResultsRead();
569 		} catch (InterruptedException e) {
570 			// Thread was interrupted; just process the results we have.
571 			// Let caller detect and deal with interruption.
572 		}
573 		int sum = 0;
574 		for (DocResult result: results) {
575 			sum += ((HitPropValueInt)numProp.get(result)).getValue();
576 		}
577 		return sum;
578 	}
