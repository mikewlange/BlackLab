8   	/**
9   	 * Compare two arrays of ints by comparing each element in succession.
10  	 *
11  	 * The first difference encountered determines the result. If the
12  	 * arrays are of different length but otherwise equal, the longest
13  	 * array will be ordered after the shorter.
14  	 *
15  	 * @param a first array
16  	 * @param b second array
17  	 * @return 0 if equal, negative if a &lt; b, positive if a &gt; b
18  	 */
19  	public static int compareArrays(int[] a, int[] b) {
20  		int n = a.length;
21  		if (b.length < n)
22  			n = b.length;
23  		for (int i = 0; i < n; i++) {
24  			if (a[i] != b[i]) {
25  				return a[i] - b[i];
26  			}
27  		}
28  		if (a.length == b.length) {
29  			// Arrays are exactly equal
30  			return 0;
31  		}
32  		if (n == a.length) {
33  			// Array b is longer than a; sort it after a
34  			return -1;
35  		}
36  		// a longer than b
37  		return 1;
38  	}

40  	/**
41  	 * Compare two arrays by comparing each element in succession.
42  	 *
43  	 * The first difference encountered determines the result. If the
44  	 * arrays are of different length but otherwise equal, the longest
45  	 * array will be ordered after the shorter.
46  	 *
47  	 * If the elements of the array are themselves arrays, this function
48  	 * is called recursively.
49  	 *
50  	 * @param a first array
51  	 * @param b second array
52  	 * @return 0 if equal, negative if a &lt; b, positive if a &gt; b
53  	 */
54  	@SuppressWarnings({ "unchecked", "rawtypes" })
55  	public static int compareArrays(Object[] a, Object[] b) {
56  		int n = a.length;
57  		if (b.length < n)
58  			n = b.length;
59  		for (int i = 0; i < n; i++) {
60  			int cmp;
61  
62  			// Figure out how to compare these two elements
63  			if (a[i] instanceof int[]) {
64  				// Use int array compare
65  				cmp = compareArrays((int[])a[i], (int[])b[i]);
66  			} else if (a[i] instanceof Object[]) {
67  				// Use Object array compare
68  				cmp = compareArrays((Object[])a[i], (Object[])b[i]);
69  			} else if (a[i] instanceof Comparable) {
70  				// Assume Comparable and use that
71  				cmp = ((Comparable)a[i]).compareTo(b[i]);
72  			} else {
73  				throw new RuntimeException("Cannot compare objects of type " + a[i].getClass());
74  			}
75  
76  			// Did that decide the comparison?
77  			if (cmp != 0) {
78  				return cmp; // yep, done
79  			}
80  		}
81  		if (a.length == b.length) {
82  			// Arrays are exactly equal
83  			return 0;
84  		}
85  		if (n == a.length) {
86  			// Array b is longer than a; sort it after a
87  			return -1;
88  		}
89  		// a longer than b
90  		return 1;
91  	}
