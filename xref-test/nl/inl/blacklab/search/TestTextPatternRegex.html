18  import org.junit.Assert;
19  import org.junit.Test;
20  
21  public class TestTextPatternRegex {
22  
23  	@Test
24  	public void testEmptyPattern() {
25  		TextPattern r = new TextPatternRegex("");
26  		Assert.assertEquals("", ((TextPatternRegex)r).getValue());
27  
28  		Assert.assertTrue(r instanceof TextPatternTerm);
29  		Assert.assertEquals("", ((TextPatternTerm)r).getValue());
30  	}
31  
32  	@Test
33  	public void testBasicPattern() {
34  		TextPattern r = new TextPatternRegex("bla");
35  		Assert.assertEquals("bla", ((TextPatternRegex)r).getValue());
36  	}
37  }
