26  import nl.inl.blacklab.TestIndex;
27  import nl.inl.blacklab.queryParser.corpusql.CorpusQueryLanguageParser;
28  import nl.inl.blacklab.queryParser.corpusql.ParseException;
29  import nl.inl.blacklab.search.QueryExplanation;
30  import nl.inl.blacklab.search.Searcher;
31  import nl.inl.blacklab.search.TextPattern;
32  import nl.inl.blacklab.search.lucene.BLSpanQuery;
33  import nl.inl.blacklab.search.lucene.optimize.ClauseCombinerNfa;
34  
35  public class TestQueryRewriteNfa {
36  
37  	static TestIndex testIndex;
38  
39  	private static  Searcher searcher;
40  
41  	@BeforeClass
42  	public static void setUp() throws Exception {
43  		testIndex = new TestIndex();
44  		searcher = testIndex.getSearcher();
45  		ClauseCombinerNfa.setNfaThreshold(ClauseCombinerNfa.MAX_NFA_MATCHING);
46  	}
47  
48  	@AfterClass
49  	public static void tearDown() {
50  		ClauseCombinerNfa.setNfaThreshold(ClauseCombinerNfa.DEFAULT_NFA_THRESHOLD);
51  		searcher.close();
52  		testIndex.close();
53  	}
54  
55  	static TextPattern getPatternFromCql(String cqlQuery) {
56  		try {
57  			cqlQuery = cqlQuery.replaceAll("'", "\""); // makes queries more readable in tests
58  			CorpusQueryLanguageParser parser = new CorpusQueryLanguageParser(new StringReader(
59  					cqlQuery));
60  			return parser.query();
61  		} catch (ParseException e) {
62  			throw new RuntimeException(e);
63  		}
64  	}
65  
66  	void assertNoRewrite(String cql, String result) {
67  		assertRewrite(cql, result, result);
68  	}
69  
70  	void assertRewrite(String cql, String before, String after) {
71  		QueryExplanation explanation = searcher.explain(getPatternFromCql(cql));
72  		if (before != null) {
73  			BLSpanQuery original = explanation.getOriginalQuery();
74  			Assert.assertEquals(before, original.toString());
75  		}
76  		BLSpanQuery rewritten = explanation.getRewrittenQuery();
77  		Assert.assertEquals(after, rewritten.toString());
78  	}
79  
80  	void assertRewriteResult(String cql, String after) {
81  		assertRewrite(cql, null, after);
82  	}
83  
84  	@Test
85  	public void testRewrite() {
86  		assertRewriteResult("\"the\" [word=\"quick\" & lemma=\"quick\"] [lemma=\"brown\"]",
87  			"FISEQ(FISEQ(AND(TERM(contents%word@i:quick), TERM(contents%lemma@i:quick)), NFA:#1:TOKEN(the,DANGLING), -1), NFA:#1:TOKEN(brown,DANGLING), 1)");
88  	}
89  
90  	@Ignore // hard to test properly with tiny indices
91  	@Test
92  	public void testRewriteSuffix() {
93  		assertRewriteResult("\"noot\" \".*p\"",
94  			"FISEQ(TERM(contents%word@i:noot), NFA:#1:REGEX(^.*p$,DANGLING), 1)");
95  	}
96  
97  	@Test
98  	public void testRewritePrefix() {
99  		assertRewriteResult("\"a.*\" \"b.*\" \"c.*\"",
100 			"FISEQ(TERM(contents%word@i:aap), NFA:#1:REGEX(^b.*$,#2:REGEX(^c.*$,DANGLING)), 1)");
101 	}

103 }
