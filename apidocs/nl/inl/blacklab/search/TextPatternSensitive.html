    Cloneable

    public class TextPatternSensitive
extends TextPattern
    TextPattern for wrapping another TextPattern so that it applies to a certain word property. For example, to find lemmas starting with "bla": TextPattern tp = new TextPatternProperty("lemma", new TextPatternWildcard("bla*"));
